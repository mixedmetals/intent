import React from 'react';

# Accordion

Collapsible sections for organizing content.

export const Accordion = ({ items }) => {
  const [openId, setOpenId] = React.useState(null)
  return (
    <div style={{ border: '1px solid var(--intent-color-border)', borderRadius: '8px', overflow: 'hidden' }}>
      {items.map((item) => (
        <div key={item.id} style={{ borderBottom: '1px solid var(--intent-color-border)' }}>
          <button
            onClick={() => setOpenId(openId === item.id ? null : item.id)}
            style={{
              width: '100%',
              padding: '12px 16px',
              textAlign: 'left',
              background: 'var(--intent-color-surface)',
              border: 'none',
              cursor: 'pointer',
              fontWeight: 500,
              display: 'flex',
              justifyContent: 'space-between',
              alignItems: 'center'
            }}
          >
            {item.title}
            <span style={{ transform: openId === item.id ? 'rotate(180deg)' : 'rotate(0)', transition: 'transform 200ms' }}>â–¼</span>
          </button>
          {openId === item.id && (
            <div style={{ padding: '12px 16px', background: 'var(--intent-color-surface-elevated)' }}>
              {item.content}
            </div>
          )}
        </div>
      ))}
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `items` | `Array<{id: string, title: string, content: ReactNode}>` | `[]` | Accordion items |

import React from 'react';

# Accordion

Collapsible sections for organizing content.

export const Accordion = ({ items }) => {
  const [openId, setOpenId] = React.useState(null)
  return (
    <div className="intent-accordion">
      {items.map((item) => (
        <div key={item.id} className={`intent-accordion__item ${openId === item.id ? 'intent-accordion__item--open' : ''}`}>
          <button
            className="intent-accordion__trigger"
            onClick={() => setOpenId(openId === item.id ? null : item.id)}
          >
            {item.title}
            <svg className="intent-accordion__chevron" viewBox="0 0 16 16" fill="none" stroke="currentColor" strokeWidth="2">
              <path d="M4 6l4 4 4-4" />
            </svg>
          </button>
          {openId === item.id && (
            <div className="intent-accordion__content">
              {item.content}
            </div>
          )}
        </div>
      ))}
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `items` | `Array<{id: string, title: string, content: ReactNode}>` | `[]` | Accordion items |

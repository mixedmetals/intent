# AspectRatio

Maintain consistent aspect ratios for media.

export const AspectRatio = ({ ratio = 16 / 9, children }) => {
  const [width, height] = ratio.toString().includes('/') 
    ? ratio.toString().split('/').map(Number)
    : [16, 9]
  const paddingBottom = `${(height / width) * 100}%`
  
  return (
    <div style={{ position: 'relative', width: '100%' }}>
      <div style={{ paddingBottom }} />
      <div style={{ 
        position: 'absolute', 
        inset: 0, 
        background: 'var(--intent-color-surface-elevated)',
        borderRadius: '8px',
        overflow: 'hidden'
      }}>
        {children || (
          <div style={{ 
            width: '100%', 
            height: '100%', 
            display: 'flex', 
            alignItems: 'center', 
            justifyContent: 'center',
            color: 'var(--intent-color-muted)'
          }}>
            {width}:{height} Aspect Ratio
          </div>
        )}
      </div>
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `ratio` | `number` | `16/9` | Aspect ratio (width/height) |
| `children` | `ReactNode` | - | Content to display |

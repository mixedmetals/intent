import React from 'react';

# Aspect

Maintain aspect ratios for responsive media.

export const Aspect = ({ ratio = 16 / 9, children }) => {
  const paddingBottom = `${(1 / ratio) * 100}%`
  
  return (
    <div style={{ position: 'relative', width: '100%' }}>
      <div style={{ paddingBottom }} />
      <div style={{ 
        position: 'absolute', 
        inset: 0, 
        overflow: 'hidden',
        borderRadius: '8px'
      }}>
        {children || (
          <div style={{ 
            width: '100%', 
            height: '100%', 
            display: 'flex', 
            alignItems: 'center', 
            justifyContent: 'center',
            background: 'linear-gradient(135deg, #667eea 0%, #764ba2 100%)',
            color: 'white',
            fontWeight: 600
          }}>
            {ratio.toFixed(2)}:1 Ratio
          </div>
        )}
      </div>
    </div>
  )
}

## Examples

### Video (16:9)

<div className="intent-demo" style={{ maxWidth: '400px' }}>
  <Aspect ratio={16 / 9} />
</div>

```tsx
<Aspect ratio={16 / 9}>
  <video src="..." />
</Aspect>
```

### Square (1:1)

<div className="intent-demo" style={{ maxWidth: '200px' }}>

</div>

```tsx

</Aspect>
```

### Portrait (4:5)

<div className="intent-demo" style={{ maxWidth: '200px' }}>

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `ratio` | `number` | `16/9` | Aspect ratio (width/height) |
| `children` | `ReactNode` | - | Content to display |

  inset: 0;

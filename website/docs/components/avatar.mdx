# Avatar

User profile image with fallback initials.

export const Avatar = ({ src, initials, size = 'md' }) => {
  const sizes = { xs: 24, sm: 32, md: 40, lg: 48, xl: 56, '2xl': 64 }
  const dimension = sizes[size] || 40
  
  return (
    <div style={{
      width: dimension,
      height: dimension,
      borderRadius: '50%',
      background: src ? 'transparent' : 'var(--intent-color-primary)',
      display: 'flex',
      alignItems: 'center',
      justifyContent: 'center',
      overflow: 'hidden',
      fontSize: dimension * 0.4,
      fontWeight: 600,
      color: 'white'
    }}>
      {src ? (
        <img src={src} alt="Avatar" style={{ width: '100%', height: '100%', objectFit: 'cover' }} />
      ) : (
        initials?.slice(0, 2).toUpperCase()
      )}
    </div>
  )
}

export const Stack = ({ children, direction = 'row', gap = 'normal', align }) => (
  <div className={`intent-stack intent-stack--direction-${direction} intent-stack--gap-${gap} ${align ? `intent-stack--align-${align}` : ''}`}>
    {children}
  </div>
)

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `src` | `string` | - | Image URL |
| `alt` | `string` | - | Alt text |
| `initials` | `string` | - | Fallback initials (2 chars max) |
| `size` | `'xs'` \| `'sm'` \| `'md'` \| `'lg'` \| `'xl'` \| `'2xl'` | `'md'` | Avatar size |

## Size Reference

# Badge

Small status indicators for UI elements.

export const Badge = ({ children, variant = 'default', size = 'medium', pill, dot }) => (
  <span className={`intent-badge intent-badge--variant-${variant} intent-badge--size-${size} ${pill ? 'intent-badge--pill' : ''}`}>
    {dot && <span className="intent-badge__dot" />}
    {children}
  </span>
)

export const Stack = ({ children, direction = 'row', gap = 'normal', align }) => (
  <div className={`intent-stack intent-stack--direction-${direction} intent-stack--gap-${gap} ${align ? `intent-stack--align-${align}` : ''}`}>
    {children}
  </div>
)

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `variant` | `'default'` \| `'primary'` \| `'success'` \| `'warning'` \| `'danger'` | `'default'` | Color variant |
| `size` | `'small'` \| `'medium'` \| `'large'` | `'medium'` | Badge size |
| `pill` | `boolean` | `false` | Rounded pill style |
| `dot` | `boolean` | `false` | Show status dot |

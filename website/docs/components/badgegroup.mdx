import React from 'react';

# BadgeGroup

Group of badges with overflow handling.

export const BadgeGroup = ({ badges, limit, removable, onRemove }) => {
  const [removed, setRemoved] = React.useState(new Set())
  
  const visibleBadges = badges
    .filter(b => !removed.has(b.id))
    .slice(0, limit || badges.length)
  
  const remaining = badges.length - visibleBadges.length
  
  const handleRemove = (id) => {
    setRemoved(prev => new Set([...prev, id]))
    onRemove?.(id)
  }
  
  const variantStyles = {
    primary: { bg: 'var(--intent-color-primary)', color: 'white' },
    success: { bg: 'var(--intent-color-success, #22c55e)', color: 'white' },
    warning: { bg: 'var(--intent-color-warning, #f59e0b)', color: 'white' },
    error: { bg: 'var(--intent-color-error, #ef4444)', color: 'white' },
    secondary: { bg: 'var(--intent-color-surface-elevated)', color: 'var(--intent-color-foreground)' }
  }
  
  return (
    <div style={{ display: 'flex', flexWrap: 'wrap', gap: '8px', alignItems: 'center' }}>
      {visibleBadges.map(badge => {
        const style = variantStyles[badge.variant] || variantStyles.secondary
        return (
          <span
            key={badge.id}
            style={{
              display: 'inline-flex',
              alignItems: 'center',
              gap: '4px',
              padding: '4px 12px',
              borderRadius: '9999px',
              fontSize: '0.75rem',
              fontWeight: 600,
              background: style.bg,
              color: style.color
            }}
          >
            {badge.label}
            {removable && (
              <button
                onClick={() => handleRemove(badge.id)}
                style={{
                  display: 'inline-flex',
                  alignItems: 'center',
                  justifyContent: 'center',
                  width: '14px',
                  height: '14px',
                  marginLeft: '2px',
                  border: 'none',
                  borderRadius: '50%',
                  background: 'rgba(255,255,255,0.3)',
                  color: 'inherit',
                  cursor: 'pointer',
                  fontSize: '10px'
                }}
              >
                Ã—
              </button>
            )}
          </span>
        )
      })}
      {remaining > 0 && (
        <span style={{
          padding: '4px 12px',
          borderRadius: '9999px',
          fontSize: '0.75rem',
          fontWeight: 600,
          background: 'var(--intent-color-surface-elevated)',
          color: 'var(--intent-color-muted)'
        }}>
          +{remaining}
        </span>
      )}
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `badges` | `Array<{id: string, label: string, variant?: string}>` | `[]` | Array of badges |
| `limit` | `number` | - | Max badges to show |
| `removable` | `boolean` | `false` | Show remove button |
| `onRemove` | `(id: string) => void` | - | Remove handler |

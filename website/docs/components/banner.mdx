import React from 'react';

# Banner

Site-wide announcement banners.

export const Banner = ({ children, variant = 'info', dismissible = false }) => {
  const [dismissed, setDismissed] = React.useState(false)
  
  if (dismissed) return null
  
  const variants = {
    info: { background: '#dbeafe', color: '#1e40af', border: '#93c5fd' },
    warning: { background: '#fef3c7', color: '#92400e', border: '#fcd34d' },
    error: { background: '#fee2e2', color: '#991b1b', border: '#fca5a5' },
    success: { background: '#dcfce7', color: '#166534', border: '#86efac' }
  }
  
  const v = variants[variant]
  
  return (
    <div style={{ 
      padding: '12px 16px', 
      background: v.background, 
      color: v.color,
      borderBottom: `1px solid ${v.border}`,
      display: 'flex',
      alignItems: 'center',
      justifyContent: 'center',
      gap: '12px',
      fontSize: '0.875rem'
    }}>
      <span style={{ flex: 1, textAlign: 'center' }}>{children}</span>
      {dismissible && (
        <button 
          onClick={() => setDismissed(true)}
          style={{ 
            background: 'none', 
            border: 'none', 
            cursor: 'pointer', 
            color: v.color,
            fontSize: '1.25rem',
            lineHeight: 1
          }}
        >
          Ã—
        </button>
      )}
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `variant` | `'info' \| 'warning' \| 'error' \| 'success'` | `'info'` | Banner style |
| `dismissible` | `boolean` | `false` | Show close button |
| `children` | `ReactNode` | - | Banner content |

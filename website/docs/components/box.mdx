# Box

Single element wrapper for spacing, borders, and backgrounds.

export const Box = ({ children, padding, background, border, borderRadius, style }) => {
  const paddingMap = { compact: '4px', normal: '8px', relaxed: '16px', loose: '24px' }
  const bgMap = { 
    surface: 'var(--intent-color-surface)', 
    'surface-elevated': 'var(--intent-color-surface-elevated)',
    transparent: 'transparent'
  }
  
  return (
    <div style={{
      padding: padding ? paddingMap[padding] : undefined,
      background: background ? bgMap[background] : undefined,
      border: border ? '1px solid var(--intent-color-border)' : undefined,
      borderRadius: borderRadius === 'sm' ? '4px' : borderRadius === 'md' ? '8px' : borderRadius === 'lg' ? '12px' : undefined,
      ...style
    }}>
      {children}
    </div>
  )
}

export const Text = ({ children }) => <span className="intent-text">{children}</span>

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `padding` | `'compact'` \| `'normal'` \| `'relaxed'` \| `'loose'` | - | Inner padding |
| `background` | `'surface'` \| `'surface-elevated'` \| `'transparent'` | - | Background color |
| `border` | `boolean` | `false` | Show border |
| `borderRadius` | `'sm'` \| `'md'` \| `'lg'` | - | Border radius |

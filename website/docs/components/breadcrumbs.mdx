# Breadcrumbs

Navigation breadcrumbs with enhanced styling.

export const Breadcrumbs = ({ items, separator = '>' }) => (
  <nav aria-label="Breadcrumb">
    <ol style={{ display: 'flex', alignItems: 'center', flexWrap: 'wrap', gap: '8px', listStyle: 'none', padding: 0, margin: 0 }}>
      {items.map((item, i) => (
        <li key={i} style={{ display: 'flex', alignItems: 'center', gap: '8px' }}>
          {i > 0 && (
            <span style={{ color: 'var(--intent-color-muted)', fontSize: '0.875rem' }}>{separator}</span>
          )}
          {item.href ? (
            <a 
              href={item.href} 
              style={{ 
                color: 'var(--intent-color-primary)', 
                textDecoration: 'none',
                fontSize: '0.875rem',
                fontWeight: 500
              }}
            >
              {item.icon && <span style={{ marginRight: '4px' }}>{item.icon}</span>}
              {item.label}
            </a>
          ) : (
            <span style={{ color: 'var(--intent-color-muted)', fontSize: '0.875rem', fontWeight: 500 }}>
              {item.icon && <span style={{ marginRight: '4px' }}>{item.icon}</span>}
              {item.label}
            </span>
          )}
        </li>
      ))}
    </ol>
  </nav>
)

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `items` | `Array<{label: string, href?: string, icon?: ReactNode}>` | `[]` | Breadcrumb items |
| `separator` | `ReactNode` | `'>'` | Custom separator |

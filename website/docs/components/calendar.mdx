# Calendar

Date picker calendar view.

## Example

<CalendarDemo />

```tsx
import { useState } from 'react'
import { Calendar } from 'intent-react'

function Example() {
  const [date, setDate] = useState(new Date())
  
  return (
    <Calendar 
      value={date}
      onChange={setDate}
    />
  )
}
```

## Range Selection

```tsx
const [range, setRange] = useState({ start: null, end: null })

<Calendar 
  mode="range"
  value={range}
  onChange={setRange}
/>
```

## Multiple Selection

```tsx
const [dates, setDates] = useState([])

<Calendar 
  mode="multiple"
  value={dates}
  onChange={setDates}
/>
```

## With Min/Max Dates

```tsx
<Calendar 
  value={date}
  onChange={setDate}
  minDate={new Date()}
  maxDate={new Date(2024, 11, 31)}
/>
```

## Disabled Dates

```tsx
<Calendar 
  value={date}
  onChange={setDate}
  disabledDates={[
    new Date(2024, 0, 1),  // New Year
    new Date(2024, 11, 25), // Christmas
  ]}
/>
```

## With Events

```tsx
<Calendar 
  value={date}
  onChange={setDate}
  events={[
    { date: new Date(2024, 0, 15), label: 'Meeting' },
    { date: new Date(2024, 0, 20), label: 'Deadline' },
  ]}
/>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `Date` \| `DateRange` \| `Date[]` | required | Selected date(s) |
| `onChange` | `(value) => void` | required | Change callback |
| `mode` | `'single'` \| `'range'` \| `'multiple'` | `'single'` | Selection mode |
| `minDate` | `Date` | - | Minimum selectable date |
| `maxDate` | `Date` | - | Maximum selectable date |
| `disabledDates` | `Date[]` | - | Disabled dates |
| `events` | `CalendarEvent[]` | - | Calendar events |
| `firstDayOfWeek` | `0` \| `1` \| `2` \| `3` \| `4` \| `5` \| `6` | `0` | First day (0=Sunday) |
| `showWeekNumbers` | `boolean` | `false` | Show week numbers |
| `locale` | `string` | `'en-US'` | Locale for formatting |

### CalendarEvent Type

```typescript
interface CalendarEvent {
  date: Date
  label: string
  color?: 'default' | 'primary' | 'success' | 'warning' | 'danger'
}
```

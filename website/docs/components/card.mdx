import React from 'react';

# Card

Content container with elevation and padding options.

export const Card = ({ children, elevation = 'medium', padding = 'normal', interactive, onClick }) => (
  <div 
    className={`intent-card intent-card--elevation-${elevation} intent-card--padding-${padding} ${interactive ? 'intent-card--interactive' : ''}`}
    onClick={onClick}
    style={interactive ? { cursor: 'pointer' } : {}}
  >
    {children}
  </div>
)

export const Stack = ({ children, direction = 'column', gap = 'normal' }) => (
  <div className={`intent-stack intent-stack--direction-${direction} intent-stack--gap-${gap}`}>
    {children}
  </div>
)

export const Text = ({ children, weight, size = 'base' }) => (
  <span className={`intent-text intent-text--size-${size} ${weight ? `intent-text--weight-${weight}` : ''}`}>
    {children}
  </span>
)

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `elevation` | `'none'` \| `'low'` \| `'medium'` \| `'high'` | `'medium'` | Shadow depth |
| `padding` | `'none'` \| `'compact'` \| `'normal'` \| `'relaxed'` | `'normal'` | Inner padding |
| `interactive` | `boolean` | `false` | Enable hover/focus states |
| `onClick` | `() => void` | - | Click handler |

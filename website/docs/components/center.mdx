import React from 'react';

# Center

Center content horizontally and vertically.

export const Center = ({ children }) => (
  <div style={{ 
    display: 'flex', 
    alignItems: 'center', 
    justifyContent: 'center',
    minHeight: '120px',
    padding: '24px'
  }}>
    {children || (
      <div style={{ 
        padding: '16px 32px', 
        background: 'var(--intent-color-primary)', 
        color: 'white',
        borderRadius: '8px',
        fontWeight: 600
      }}>
        Centered Content
      </div>
    )}
  </div>
)

## Examples

<div className="intent-demo">

    </div>
  </Center>
</div>

```tsx
<Center>
  <div>Centered content</div>
</Center>
```

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | Content to center |

import React from 'react';
import { Checkbox, Stack } from 'intent-react';

# Checkbox

Boolean toggle for multiple selections. Uses the **Milled** archetype with hairline borders and tactile feedback.

export const CheckboxDemo = ({ label, defaultChecked = false, disabled = false, size = 'md' }) => {
  const [checked, setChecked] = React.useState(defaultChecked);
  
  return (
    <Checkbox
      label={label}
      checked={checked}
      onChange={(e) => setChecked(e.target.checked)}
      disabled={disabled}
      size={size}
    />
  );
};

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `size` | `'sm' \| 'md' \| 'lg'` | `'md'` | Checkbox size |
| `variant` | `'default' \| 'filled' \| 'outline'` | `'default'` | Visual variant |
| `checked` | `boolean` | - | Controlled checked state |
| `defaultChecked` | `boolean` | - | Uncontrolled default state |
| `disabled` | `boolean` | `false` | Disable interaction |
| `indeterminate` | `boolean` | `false` | Indeterminate state (parent checkbox) |
| `label` | `string` | - | Label text |
| `onChange` | `(checked: boolean) => void` | - | Change handler |

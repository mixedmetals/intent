# Checkbox

Boolean toggle for multiple selections.

export const Checkbox = ({ label, description, size = 'medium', checked, disabled }) => (
  <label className="intent-checkbox">
    <input 
      type="checkbox"
      className="intent-checkbox__input"
      checked={checked}
      disabled={disabled}
    />
    <div>
      <div className="intent-checkbox__label">{label}</div>
      {description && <div style={{ fontSize: '14px', color: 'var(--intent-color-text-muted)', marginTop: '2px' }}>{description}</div>}
    </div>
  </label>
)

export const Stack = ({ children, gap = 'normal' }) => (
  <div className={`intent-stack intent-stack--direction-column intent-stack--gap-${gap}`}>
    {children}
  </div>
)

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `label` | `string` | - | Checkbox label |
| `description` | `string` | - | Helper text |
| `checked` | `boolean` | - | Controlled checked state |
| `disabled` | `boolean` | `false` | Disable checkbox |
| `onChange` | `(e: ChangeEvent) => void` | - | Change handler |

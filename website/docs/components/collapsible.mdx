import React from 'react';

# Collapsible

Expandable and collapsible content sections. Uses the **Elevated** archetype for the container.

export const Collapsible = ({ title, children, defaultOpen = false }) => {
  const [isOpen, setIsOpen] = React.useState(defaultOpen)
  return (
    <div style={{ border: '1px solid var(--intent-color-border)', borderRadius: '8px', overflow: 'hidden' }}>
      <button
        onClick={() => setIsOpen(!isOpen)}
        style={{
          width: '100%',
          padding: '12px 16px',
          textAlign: 'left',
          background: 'var(--intent-color-surface)',
          border: 'none',
          cursor: 'pointer',
          fontWeight: 500,
          display: 'flex',
          justifyContent: 'space-between',
          alignItems: 'center'
        }}
      >
        {title}
        <span style={{ 
          transform: isOpen ? 'rotate(180deg)' : 'rotate(0)', 
          transition: 'transform 200ms' 
        }}>â–¼</span>
      </button>
      <div style={{ 
        maxHeight: isOpen ? '500px' : '0',
        overflow: 'hidden',
        transition: 'max-height 200ms ease-out'
      }}>
        <div style={{ padding: '16px', borderTop: '1px solid var(--intent-color-border)', background: 'var(--intent-color-surface-elevated)' }}>
          {children}
        </div>
      </div>
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `title` | `ReactNode` | - | Section title |
| `defaultOpen` | `boolean` | `false` | Initial open state |
| `open` | `boolean` | - | Controlled open state |
| `onOpenChange` | `(open: boolean) => void` | - | Open change callback |

## Design Notes

- Header uses **Flat** archetype
- Content area uses **Elevated** background
- Smooth height animation
- Chevron rotates when expanded
- Common in FAQ and settings sections

import React from 'react';

# ColorPicker

Input for selecting colors.

export const ColorPicker = ({ defaultValue = '#3b82f6', label }) => {
  const [color, setColor] = React.useState(defaultValue)
  const [isOpen, setIsOpen] = React.useState(false)
  
  const presetColors = ['#ef4444', '#f97316', '#f59e0b', '#84cc16', '#22c55e', '#10b981', '#14b8a6', '#06b6d4', '#0ea5e9', '#3b82f6', '#6366f1', '#8b5cf6', '#a855f7', '#d946ef', '#ec4899', '#f43f5e']
  
  return (
    <div style={{ width: '100%', maxWidth: '280px' }}>
      {label && <label style={{ display: 'block', fontSize: '0.875rem', fontWeight: 500, marginBottom: '8px' }}>{label}</label>}
      <div style={{ position: 'relative' }}>
        <button
          onClick={() => setIsOpen(!isOpen)}
          style={{
            display: 'flex',
            alignItems: 'center',
            gap: '12px',
            width: '100%',
            padding: '10px 12px',
            border: '1px solid var(--intent-color-border)',
            borderRadius: '8px',
            background: 'var(--intent-color-surface)',
            cursor: 'pointer'
          }}
        >
          <div style={{ width: '24px', height: '24px', borderRadius: '4px', background: color, border: '1px solid var(--intent-color-border)' }} />
          <span style={{ fontSize: '0.875rem', fontFamily: 'monospace', textTransform: 'uppercase' }}>{color}</span>
        </button>
        
        {isOpen && (
          <div style={{
            position: 'absolute',
            top: '100%',
            left: 0,
            marginTop: '8px',
            padding: '16px',
            background: 'var(--intent-color-surface)',
            border: '1px solid var(--intent-color-border)',
            borderRadius: '12px',
            boxShadow: '0 10px 15px -3px rgba(0,0,0,0.1)',
            zIndex: 50
          }}>
            <div style={{ display: 'grid', gridTemplateColumns: 'repeat(8, 1fr)', gap: '6px', marginBottom: '12px' }}>
              {presetColors.map(c => (
                <button
                  key={c}
                  onClick={() => setColor(c)}
                  style={{
                    width: '24px',
                    height: '24px',
                    borderRadius: '4px',
                    background: c,
                    border: color === c ? '2px solid var(--intent-color-foreground)' : '1px solid var(--intent-color-border)',
                    cursor: 'pointer'
                  }}
                />
              ))}
            </div>
            <input
              type="color"
              value={color}
              onChange={(e) => setColor(e.target.value)}
              style={{ width: '100%', height: '40px', border: 'none', borderRadius: '6px', cursor: 'pointer' }}
            />
          </div>
        )}
      </div>
    </div>
  )
}

## Examples

### Basic ColorPicker

<div className="intent-demo">
  <ColorPicker defaultValue="#3b82f6" label="Choose Color" />
</div>

```tsx
<ColorPicker 
  defaultValue="#3b82f6"
  onChange={(color) => console.log(color)}
/>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | - | Controlled color value |
| `defaultValue` | `string` | `'#000000'` | Default color |
| `label` | `string` | - | Input label |
| `onChange` | `(color: string) => void` | - | Color change handler |
| `presets` | `string[]` | - | Preset color options |

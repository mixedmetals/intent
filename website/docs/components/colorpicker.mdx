# ColorPicker

Input for selecting colors.

## Examples

### Basic ColorPicker

```tsx
import { useState } from 'react'
import { ColorPicker } from 'intent-react'

function Example() {
  const [color, setColor] = useState('#3b82f6')
  
  return (
    <ColorPicker 
      value={color}
      onChange={setColor}
    />
  )
}
```

### With Presets

```tsx
<ColorPicker 
  value={color}
  onChange={setColor}
  presets={[
    '#ef4444', '#f97316', '#f59e0b', '#84cc16',
    '#10b981', '#06b6d4', '#3b82f6', '#8b5cf6',
  ]}
/>
```

### Different Formats

```tsx
<ColorPicker 
  value={color}
  onChange={setColor}
  format="hex"     // #3b82f6
/>

<ColorPicker 
  value={color}
  onChange={setColor}
  format="rgb"     // rgb(59, 130, 246)
/>

<ColorPicker 
  value={color}
  onChange={setColor}
  format="hsl"     // hsl(217, 91%, 60%)
/>
```

### Alpha Support

```tsx
<ColorPicker 
  value={color}
  onChange={setColor}
  alpha
/>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | - | Selected color |
| `onChange` | `(color: string) => void` | - | Change callback |
| `format` | `'hex'` \| `'rgb'` \| `'hsl'` | `'hex'` | Color format |
| `alpha` | `boolean` | `false` | Allow alpha selection |
| `presets` | `string[]` | - | Preset colors |
| `disabled` | `boolean` | `false` | Disable picker |

# Combobox

Autocomplete input with dropdown suggestions.

## Examples

### Basic Combobox

```tsx
import { useState } from 'react'
import { Combobox } from 'intent-react'

const users = [
  { id: '1', name: 'John Doe', email: 'john@example.com' },
  { id: '2', name: 'Jane Smith', email: 'jane@example.com' },
  { id: '3', name: 'Bob Johnson', email: 'bob@example.com' },
]

function Example() {
  const [selected, setSelected] = useState(null)
  
  return (
    <Combobox
      options={users}
      value={selected}
      onChange={setSelected}
      getLabel={(user) => user.name}
      placeholder="Select a user..."
    />
  )
}
```

### Async Search

```tsx
<Combobox
  loadOptions={async (query) => {
    const response = await fetch(`/api/users?q=${query}`)
    return response.json()
  }}
  placeholder="Search users..."
/>
```

### Multi-select

```tsx
const [selected, setSelected] = useState([])

<Combobox
  multiple
  options={users}
  value={selected}
  onChange={setSelected}
  getLabel={(user) => user.name}
/>
```

### With Custom Item

```tsx
<Combobox
  options={users}
  getLabel={(user) => user.name}
  renderItem={(user) => (
    <div>
      <div>{user.name}</div>
      <div style={{ fontSize: 12, color: 'gray' }}>{user.email}</div>
    </div>
  )}
/>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `options` | `T[]` | `[]` | Available options |
| `value` | `T` \| `T[]` | - | Selected value(s) |
| `onChange` | `(value) => void` | - | Change callback |
| `getLabel` | `(option: T) => string` | - | Extract label |
| `getValue` | `(option: T) => string` | - | Extract value |
| `multiple` | `boolean` | `false` | Allow multiple selection |
| `loadOptions` | `(query: string) => Promise<T[]>` | - | Async loader |
| `placeholder` | `string` | - | Input placeholder |
| `renderItem` | `(option: T) => ReactNode` | - | Custom item render |
| `creatable` | `boolean` | `false` | Allow creating new options |

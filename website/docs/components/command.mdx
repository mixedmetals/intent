# Command

Command palette for quick actions and navigation.

## Example

```tsx
import { useState } from 'react'
import { Command, Button } from 'intent-react'

function Example() {
  const [open, setOpen] = useState(false)
  
  return (
    <>
      <Button onClick={() => setOpen(true)}>
        Open Command Palette
      </Button>
      
      <Command
        open={open}
        onOpenChange={setOpen}
        placeholder="Type a command or search..."
        groups={[
          {
            heading: 'Navigation',
            items: [
              { id: 'home', label: 'Go to Home', icon: 'home', shortcut: '⌘H' },
              { id: 'projects', label: 'Go to Projects', icon: 'folder', shortcut: '⌘P' },
              { id: 'settings', label: 'Go to Settings', icon: 'settings', shortcut: '⌘,' },
            ]
          },
          {
            heading: 'Actions',
            items: [
              { id: 'new', label: 'New Project', icon: 'plus', shortcut: '⌘N' },
              { id: 'search', label: 'Search Files', icon: 'search', shortcut: '⌘F' },
              { id: 'theme', label: 'Toggle Theme', icon: 'sun', shortcut: '⌘T' },
            ]
          },
        ]}
        onSelect={(item) => {
          console.log('Selected:', item.id)
          setOpen(false)
        }}
      />
    </>
  )
}
```

## Keyboard Shortcut

Bind to keyboard shortcut:

```tsx
import { useEffect } from 'react'

function App() {
  const [open, setOpen] = useState(false)
  
  useEffect(() => {
    const handleKeyDown = (e) => {
      if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
        e.preventDefault()
        setOpen(true)
      }
    }
    
    window.addEventListener('keydown', handleKeyDown)
    return () => window.removeEventListener('keydown', handleKeyDown)
  }, [])
  
  return <Command open={open} onOpenChange={setOpen} ... />
}
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `open` | `boolean` | required | Controlled open state |
| `onOpenChange` | `(open: boolean) => void` | - | Open state callback |
| `placeholder` | `string` | `'Search...'` | Input placeholder |
| `searchable` | `boolean` | `true` | Enable search filtering |
| `groups` | `CommandGroup[]` | `[]` | Command groups |
| `recent` | `CommandItem[]` | - | Recent items |
| `emptyMessage` | `string` | `'No results found'` | Empty state message |
| `onSelect` | `(item: CommandItem) => void` | - | Selection callback |

### CommandGroup Type

```typescript
interface CommandGroup {
  heading: string
  items: CommandItem[]
}

interface CommandItem {
  id: string
  label: string
  description?: string
  icon?: string | ReactNode
  shortcut?: string
  disabled?: boolean
  onClick?: () => void
}
```

import React from 'react';

# Command

Command palette for quick navigation and actions.

export const Command = ({ placeholder = "Type a command or search..." }) => {
  const [isOpen, setIsOpen] = React.useState(false)
  const [search, setSearch] = React.useState('')
  
  const groups = [
    { name: 'Suggestions', items: ['Calendar', 'Search Emoji', 'Calculator'] },
    { name: 'Settings', items: ['Profile', 'Billing', 'Settings', 'Log out'] }
  ]
  
  React.useEffect(() => {
    const handleKeyDown = (e) => {
      if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
        e.preventDefault()
        setIsOpen(prev => !prev)
      }
      if (e.key === 'Escape') setIsOpen(false)
    }
    window.addEventListener('keydown', handleKeyDown)
    return () => window.removeEventListener('keydown', handleKeyDown)
  }, [])
  
  if (!isOpen) return (
    <div style={{ fontSize: '0.875rem', color: 'var(--intent-color-muted)' }}>
      Press <kbd style={{ padding: '2px 6px', background: 'var(--intent-color-surface-elevated)', border: '1px solid var(--intent-color-border)', borderRadius: '4px', fontSize: '0.75rem' }}>‚åòK</kbd> to open
    </div>
  )
  
  return (
    <div style={{ position: 'fixed', inset: 0, zIndex: 50, display: 'flex', alignItems: 'flex-start', justifyContent: 'center', paddingTop: '100px' }}>
      <div style={{ position: 'absolute', inset: 0, background: 'rgba(0,0,0,0.5)' }} onClick={() => setIsOpen(false)} />
      <div style={{ position: 'relative', width: '100%', maxWidth: '640px', background: 'var(--intent-color-surface)', borderRadius: '12px', boxShadow: '0 25px 50px -12px rgba(0,0,0,0.25)', overflow: 'hidden' }}>
        <div style={{ display: 'flex', alignItems: 'center', padding: '16px', borderBottom: '1px solid var(--intent-color-border)', gap: '12px' }}>
          <span style={{ color: 'var(--intent-color-muted)' }}>üîç</span>
          <input 
            type="text" 
            value={search}
            onChange={(e) => setSearch(e.target.value)}
            placeholder={placeholder}
            style={{ flex: 1, border: 'none', background: 'none', fontSize: '1rem', outline: 'none' }}
            autoFocus
          />
          <kbd style={{ padding: '4px 8px', background: 'var(--intent-color-surface-elevated)', border: '1px solid var(--intent-color-border)', borderRadius: '4px', fontSize: '0.75rem' }}>ESC</kbd>
        </div>
        <div style={{ maxHeight: '400px', overflow: 'auto', padding: '8px' }}>
          {groups.map((group, gi) => (
            <div key={gi}>
              <div style={{ padding: '8px 12px', fontSize: '0.75rem', fontWeight: 600, color: 'var(--intent-color-muted)', textTransform: 'uppercase' }}>{group.name}</div>
              {group.items.filter(i => i.toLowerCase().includes(search.toLowerCase())).map((item, ii) => (
                <button 
                  key={ii} 
                  style={{ 
                    display: 'flex', 
                    alignItems: 'center', 
                    gap: '12px', 
                    width: '100%', 
                    padding: '10px 12px', 
                    textAlign: 'left', 
                    background: 'none', 
                    border: 'none', 
                    borderRadius: '6px', 
                    cursor: 'pointer',
                    fontSize: '0.875rem'
                  }}
                  onMouseEnter={(e) => e.target.style.background = 'var(--intent-color-surface-hover)'}
                  onMouseLeave={(e) => e.target.style.background = 'none'}
                  onClick={() => setIsOpen(false)}
                >
                  <span>‚åò</span>
                  <span>{item}</span>
                </button>
              ))}
            </div>
          ))}
        </div>
        <div style={{ display: 'flex', gap: '16px', padding: '12px 16px', borderTop: '1px solid var(--intent-color-border)', fontSize: '0.75rem', color: 'var(--intent-color-muted)' }}>
          <span><kbd>‚Üë‚Üì</kbd> to navigate</span>
          <span><kbd>‚Üµ</kbd> to select</span>
        </div>
      </div>
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | Command content |
| `label` | `string` | - | ARIA label |
| `placeholder` | `string` | `"Type a command..."` | Input placeholder |

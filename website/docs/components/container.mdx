# Container

Centered, max-width wrapper for page content.

export const Container = ({ children, size = 'medium', padding = 'normal', style }) => {
  const sizeMap = { small: '640px', medium: '768px', large: '1024px', xl: '1280px' }
  const paddingMap = { compact: '4px', normal: '16px', relaxed: '24px' }
  
  return (
    <div style={{
      maxWidth: sizeMap[size],
      margin: '0 auto',
      padding: paddingMap[padding],
      ...style
    }}>
      {children}
    </div>
  )
}

export const Card = ({ children }) => <div className="intent-card intent-card--padding-normal intent-card--elevation-low">{children}</div>
export const Text = ({ children }) => <span className="intent-text">{children}</span>

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `size` | `'small'` \| `'medium'` \| `'large'` \| `'xl'` | `'medium'` | Max width constraint |
| `padding` | `'compact'` \| `'normal'` \| `'relaxed'` | `'normal'` | Horizontal padding |

## Size Breakpoints

| Size | Max Width | Use Case |
|------|-----------|----------|
| `small` | 640px | Forms, focused content |
| `medium` | 768px | Articles, reading content |
| `large` | 1024px | Dashboards, tables |
| `xl` | 1280px | Full layouts |

import React from 'react';

# ContextMenu

Right-click context menus. Uses the **Elevated** archetype.

export const ContextMenuDemo = () => {
  const [isOpen, setIsOpen] = React.useState(false)
  const [position, setPosition] = React.useState({ x: 0, y: 0 })
  
  const items = [
    { label: 'Cut', icon: 'âœ‚ï¸', shortcut: 'âŒ˜X' },
    { label: 'Copy', icon: 'ðŸ“‹', shortcut: 'âŒ˜C' },
    { label: 'Paste', icon: 'ðŸ“Œ', shortcut: 'âŒ˜V' },
    { separator: true },
    { label: 'Delete', icon: 'ðŸ—‘ï¸', shortcut: 'âŒ«' }
  ]
  
  const handleContextMenu = (e) => {
    e.preventDefault()
    setPosition({ x: e.clientX, y: e.clientY })
    setIsOpen(true)
  }
  
  React.useEffect(() => {
    const handleClick = () => setIsOpen(false)
    if (isOpen) {
      document.addEventListener('click', handleClick)
      return () => document.removeEventListener('click', handleClick)
    }
  }, [isOpen])
  
  return (
    <>
      <div 
        onContextMenu={handleContextMenu} 
        style={{ padding: '32px', border: '2px dashed var(--intent-color-border)', borderRadius: '8px', textAlign: 'center' }}
      >
        Right-click here
      </div>
      {isOpen && (
        <div style={{
          position: 'fixed',
          left: position.x,
          top: position.y,
          minWidth: '200px',
          background: 'var(--intent-color-surface)',
          border: '1px solid var(--intent-color-border)',
          borderRadius: '8px',
          boxShadow: '0 10px 15px -3px rgba(0,0,0,0.1)',
          zIndex: 50,
          padding: '4px'
        }}>
          {items.map((item, i) => (
            item.separator ? (
              <div key={i} style={{ height: '1px', background: 'var(--intent-color-border)', margin: '4px' }} />
            ) : (
              <button
                key={i}
                style={{
                  display: 'flex',
                  alignItems: 'center',
                  gap: '12px',
                  width: '100%',
                  padding: '8px 12px',
                  textAlign: 'left',
                  background: 'none',
                  border: 'none',
                  cursor: 'pointer',
                  fontSize: '0.875rem',
                  borderRadius: '4px'
                }}
                onMouseEnter={(e) => e.currentTarget.style.background = 'var(--intent-color-surface-hover)'}
                onMouseLeave={(e) => e.currentTarget.style.background = 'none'}
              >
                {item.icon && <span>{item.icon}</span>}
                <span>{item.label}</span>
                {item.shortcut && <span style={{ marginLeft: 'auto', fontSize: '0.75rem', color: 'var(--intent-color-muted)' }}>{item.shortcut}</span>}
              </button>
            )
          ))}
        </div>
      )}
    </>
  )
}

## Props

### ContextMenu

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | Menu items |

### ContextMenuItem

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `shortcut` | `string` | - | Keyboard shortcut |
| `disabled` | `boolean` | `false` | Disabled state |
| `children` | `ReactNode` | - | Item content |

## Design Notes

- Triggered by right-click
- Positioned at cursor location
- Uses **Elevated** archetype
- Supports icons and shortcuts
- Click outside to close

# DatePicker

Input for selecting dates with calendar popup.

## Examples

### Basic DatePicker

```tsx
import { useState } from 'react'
import { DatePicker } from 'intent-react'

function Example() {
  const [date, setDate] = useState(new Date())
  
  return (
    <DatePicker 
      value={date}
      onChange={setDate}
      placeholder="Select date..."
    />
  )
}
```

### Date Range

```tsx
const [range, setRange] = useState({ start: null, end: null })

<DatePicker 
  mode="range"
  value={range}
  onChange={setRange}
/>
```

### With Min/Max

```tsx
<DatePicker 
  value={date}
  onChange={setDate}
  minDate={new Date()}
  maxDate={new Date(2024, 11, 31)}
/>
```

### Different Formats

```tsx
<DatePicker format="MM/dd/yyyy" />  // 12/25/2024
<DatePicker format="dd/MM/yyyy" />  // 25/12/2024
<DatePicker format="yyyy-MM-dd" />  // 2024-12-25
```

### With Time

```tsx
<DatePicker 
  withTime
  value={dateTime}
  onChange={setDateTime}
/>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `Date` \| `DateRange` | - | Selected date(s) |
| `onChange` | `(value) => void` | - | Change callback |
| `mode` | `'single'` \| `'range'` | `'single'` | Selection mode |
| `minDate` | `Date` | - | Minimum date |
| `maxDate` | `Date` | - | Maximum date |
| `format` | `string` | `'MMM dd, yyyy'` | Display format |
| `withTime` | `boolean` | `false` | Include time selection |
| `placeholder` | `string` | - | Input placeholder |
| `disabledDates` | `Date[]` | - | Disabled dates |

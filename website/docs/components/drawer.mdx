import { Stack, Button, Drawer, Text } from 'intent-react'

# Drawer

Slide-out panel for navigation or supplementary content.

## Example

```tsx
import { useState } from 'react'
import { Button, Drawer, Text } from 'intent-react'

function Example() {
  const [open, setOpen] = useState(false)
  
  return (
    <>
      <Button onClick={() => setOpen(true)}>
        Open Drawer
      </Button>
      
      <Drawer
        open={open}
        onOpenChange={setOpen}
        title="Drawer Title"
        position="right"
      >
        <Text>Drawer content goes here.</Text>
      </Drawer>
    </>
  )
}
```

## Positions

Drawers can slide in from any edge:

| Position | Description |
|----------|-------------|
| `left` | Left side navigation |
| `right` | Right side details |
| `top` | Top banner/bar |
| `bottom` | Bottom sheet (mobile) |

```tsx
<Drawer position="left" open={open} onOpenChange={setOpen}>
  ...
</Drawer>

<Drawer position="bottom" open={open} onOpenChange={setOpen}>
  ...
</Drawer>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `open` | `boolean` | - | Controlled open state |
| `onOpenChange` | `(open: boolean) => void` | - | State change callback |
| `title` | `string` | - | Drawer header title |
| `position` | `'left'` \| `'right'` \| `'top'` \| `'bottom'` | `'right'` | Drawer position |
| `size` | `'sm'` \| `'md'` \| `'lg'` \| `'full'` | `'md'` | Drawer size |
| `showClose` | `boolean` | `true` | Show close button |
| `closeOnOverlay` | `boolean` | `true` | Close when clicking backdrop |

## Pattern: Navigation Drawer

```tsx
function Layout() {
  const [navOpen, setNavOpen] = useState(false)
  
  return (
    <>
      <Button 
        importance="tertiary" 
        icon="menu"
        onClick={() => setNavOpen(true)}
      >
        Menu
      </Button>
      
      <Drawer
        position="left"
        size="sm"
        open={navOpen}
        onOpenChange={setNavOpen}
      >
        <nav>
          <NavLink href="/">Home</NavLink>
          <NavLink href="/products">Products</NavLink>
          <NavLink href="/about">About</NavLink>
          <NavLink href="/contact">Contact</NavLink>
        </nav>
      </Drawer>
    </>
  )
}
```

## Pattern: Details Drawer

```tsx
function ProductList() {
  const [selectedProduct, setSelectedProduct] = useState(null)
  
  return (
    <>
      {products.map(product => (
        <ProductCard 
          key={product.id} 
          product={product}
          onClick={() => setSelectedProduct(product)}
        />
      ))}
      
      <Drawer
        position="right"
        size="lg"
        open={!!selectedProduct}
        onOpenChange={() => setSelectedProduct(null)}
        title={selectedProduct?.name}
      >
        {selectedProduct && (
          <ProductDetails product={selectedProduct} />
        )}
      </Drawer>
    </>
  )
}
```

## Modal vs Drawer

Use **Drawer** when:
- Content is supplementary (not blocking)
- Space is needed for forms or lists
- Navigation on mobile/desktop
- Persistent context is important

Use **Modal** when:
- User must make a decision
- Critical information requires attention
- Blocking the main content is acceptable

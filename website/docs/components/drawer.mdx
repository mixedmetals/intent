import React from 'react';

# Drawer

Slide-out panels for navigation or forms.

export const Drawer = ({ title, children, isOpen, onClose, placement = 'right', footer }) => {
  if (!isOpen) return null
  return (
    <div style={{ position: 'fixed', inset: 0, zIndex: 50 }}>
      {/* Backdrop */}
      <div 
        style={{ 
          position: 'absolute', 
          inset: 0, 
          background: 'rgba(0,0,0,0.5)', 
          backdropFilter: 'blur(4px)' 
        }} 
        onClick={onClose}
      />
      {/* Drawer */}
      <div style={{ 
        position: 'absolute', 
        top: 0,
        bottom: 0,
        [placement]: 0,
        width: '100%',
        maxWidth: '400px',
        background: 'var(--intent-color-surface)', 
        boxShadow: placement === 'right' ? '-4px 0 25px rgba(0,0,0,0.1)' : '4px 0 25px rgba(0,0,0,0.1)',
        display: 'flex',
        flexDirection: 'column'
      }}>
        {/* Header */}
        <div style={{ padding: '16px 24px', borderBottom: '1px solid var(--intent-color-border)', display: 'flex', justifyContent: 'space-between', alignItems: 'center' }}>
          <h3 style={{ margin: 0, fontSize: '1.125rem', fontWeight: 600 }}>{title}</h3>
          <button 
            onClick={onClose}
            style={{ 
              background: 'none', 
              border: 'none', 
              fontSize: '1.5rem', 
              cursor: 'pointer',
              color: 'var(--intent-color-muted)'
            }}
          >
            Ã—
          </button>
        </div>
        {/* Body */}
        <div style={{ padding: '24px', flex: 1, overflow: 'auto' }}>
          {children}
        </div>
        {/* Footer */}
        {footer && (
          <div style={{ padding: '16px 24px', borderTop: '1px solid var(--intent-color-border)', display: 'flex', justifyContent: 'flex-end', gap: '12px' }}>
            {footer}
          </div>
        )}
      </div>
    </div>
  )
}

```tsx
  <Form>...</Form>

```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `isOpen` | `boolean` | `false` | Whether drawer is open |
| `onClose` | `() => void` | - | Called when drawer closes |
| `title` | `string` | - | Drawer title |
| `placement` | `'left' \| 'right'` | `'right'` | Drawer position |
| `children` | `ReactNode` | - | Drawer content |
| `footer` | `ReactNode` | - | Footer content |

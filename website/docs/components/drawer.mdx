import React from 'react';
import { Drawer, Button } from 'intent-react';

# Drawer

Slide-out panels for navigation or forms. Uses the **Elevated** archetype for the panel and **Milled** for header/footer elements.

export const DrawerDemo = () => {
  const [isOpen, setIsOpen] = React.useState(false)
  return (
    <>
      <Button importance="secondary" onClick={() => setIsOpen(true)}>Open Drawer</Button>
      <Drawer 
        isOpen={isOpen} 
        onClose={() => setIsOpen(false)}
        title="Settings"
        footer={
          <>
            <Button importance="secondary" onClick={() => setIsOpen(false)}>Cancel</Button>
            <Button onClick={() => setIsOpen(false)}>Save</Button>
          </>
        }
      >
        <div style={{ display: 'flex', flexDirection: 'column', gap: '16px' }}>
          <div>
            <label style={{ display: 'block', fontSize: '0.875rem', fontWeight: 500, marginBottom: '4px' }}>Display Name</label>
            <input type="text" placeholder="Enter name" style={{ width: '100%', padding: '8px 12px', border: '1px solid var(--intent-color-border)', borderRadius: '6px' }} />
          </div>
          <div>
            <label style={{ display: 'block', fontSize: '0.875rem', fontWeight: 500, marginBottom: '4px' }}>Email</label>
            <input type="email" placeholder="Enter email" style={{ width: '100%', padding: '8px 12px', border: '1px solid var(--intent-color-border)', borderRadius: '6px' }} />
          </div>
        </div>
      </Drawer>
    </>
  )
}

```tsx
const [isOpen, setIsOpen] = useState(false)

<Button onClick={() => setIsOpen(true)}>Open Drawer</Button>

<Drawer 
  isOpen={isOpen} 
  onClose={() => setIsOpen(false)}
  title="Settings"
  footer={
    <>
      <Button importance="secondary">Cancel</Button>
      <Button>Save</Button>
    </>
  }
>
  <Form>...</Form>
</Drawer>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `isOpen` | `boolean` | `false` | Whether drawer is open |
| `onClose` | `() => void` | - | Called when drawer closes |
| `title` | `string` | - | Drawer title |
| `placement` | `'left' \| 'right'` | `'right'` | Drawer position |
| `children` | `ReactNode` | - | Drawer content |
| `footer` | `ReactNode` | - | Footer content |

## Design Notes

- Drawer panel uses **Elevated** archetype (floating above content)
- Backdrop uses `rgba(0,0,0,0.5)` with `backdrop-filter: blur(4px)`
- Header/footer have 1px borders
- Shadow direction respects placement (left/right)

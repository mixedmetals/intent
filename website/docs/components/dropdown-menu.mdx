import React from 'react';

# DropdownMenu

Dropdown menus with submenus and keyboard navigation.

export const DropdownMenu = ({ trigger, items }) => {
  const [isOpen, setIsOpen] = React.useState(false)
  
  return (
    <div style={{ position: 'relative', display: 'inline-block' }}>
      <div onClick={() => setIsOpen(!isOpen)}>{trigger}</div>
      {isOpen && (
        <>
          <div style={{ position: 'fixed', inset: 0 }} onClick={() => setIsOpen(false)} />
          <div style={{
            position: 'absolute',
            top: '100%',
            left: 0,
            marginTop: '4px',
            minWidth: '220px',
            background: 'var(--intent-color-surface)',
            border: '1px solid var(--intent-color-border)',
            borderRadius: '8px',
            boxShadow: '0 10px 15px -3px rgba(0,0,0,0.1)',
            zIndex: 50,
            padding: '4px'
          }}>
            {items.map((item, i) => (
              <div key={i}>
                {item.separator ? (
                  <div style={{ height: '1px', background: 'var(--intent-color-border)', margin: '4px' }} />
                ) : (
                  <button
                    onClick={() => { item.onClick?.(); setIsOpen(false); }}
                    disabled={item.disabled}
                    style={{
                      display: 'flex',
                      alignItems: 'center',
                      gap: '12px',
                      width: '100%',
                      padding: '8px 12px',
                      textAlign: 'left',
                      background: 'none',
                      border: 'none',
                      cursor: item.disabled ? 'not-allowed' : 'pointer',
                      fontSize: '0.875rem',
                      borderRadius: '4px',
                      opacity: item.disabled ? 0.5 : 1
                    }}
                    onMouseEnter={(e) => !item.disabled && (e.target.style.background = 'var(--intent-color-surface-hover)')}
                    onMouseLeave={(e) => e.target.style.background = 'none'}
                  >
                    {item.icon && <span>{item.icon}</span>}
                    <span>{item.label}</span>
                    {item.shortcut && <span style={{ marginLeft: 'auto', fontSize: '0.75rem', color: 'var(--intent-color-muted)' }}>{item.shortcut}</span>}
                  </button>
                )}
              </div>
            ))}
          </div>
        </>
      )}
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `trigger` | `ReactNode` | - | Trigger element |
| `items` | `Array<{label?: string, icon?: ReactNode, shortcut?: string, onClick?: () => void, disabled?: boolean, separator?: boolean}>` | `[]` | Menu items |

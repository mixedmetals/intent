import React from 'react';

# Dropdown

Simple dropdown menu component.

export const Dropdown = ({ trigger, items }) => {
  const [isOpen, setIsOpen] = React.useState(false)
  const [selected, setSelected] = React.useState(null)
  
  return (
    <div style={{ position: 'relative', display: 'inline-block' }}>
      <div onClick={() => setIsOpen(!isOpen)}>{trigger}</div>
      {isOpen && (
        <>
          <div style={{ position: 'fixed', inset: 0 }} onClick={() => setIsOpen(false)} />
          <div style={{
            position: 'absolute',
            top: '100%',
            left: 0,
            marginTop: '4px',
            minWidth: '200px',
            background: 'var(--intent-color-surface)',
            border: '1px solid var(--intent-color-border)',
            borderRadius: '8px',
            boxShadow: '0 10px 15px -3px rgba(0,0,0,0.1)',
            zIndex: 50
          }}>
            {items.map((item, i) => (
              <button
                key={i}
                onClick={() => { setSelected(item.value); setIsOpen(false); item.onClick?.(); }}
                style={{
                  display: 'flex',
                  alignItems: 'center',
                  gap: '12px',
                  width: '100%',
                  padding: '10px 16px',
                  textAlign: 'left',
                  background: selected === item.value ? 'var(--intent-color-surface-hover)' : 'none',
                  border: 'none',
                  cursor: 'pointer',
                  fontSize: '0.875rem',
                  borderBottom: i < items.length - 1 ? '1px solid var(--intent-color-border)' : 'none'
                }}
              >
                {item.icon && <span>{item.icon}</span>}
                <span>{item.label}</span>
                {selected === item.value && <span style={{ marginLeft: 'auto' }}>âœ“</span>}
              </button>
            ))}
          </div>
        </>
      )}
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `trigger` | `ReactNode` | - | Dropdown trigger element |
| `items` | `Array<{label: string, value: string, icon?: ReactNode, onClick?: () => void}>` | `[]` | Dropdown items |
| `value` | `string` | - | Selected value |
| `onChange` | `(value: string) => void` | - | Selection handler |

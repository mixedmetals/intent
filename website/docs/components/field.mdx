import React from 'react';

# Field

Label and validation wrapper for form inputs.

export const Field = ({ label, hint, error, children, required }) => (
  <div style={{ display: 'flex', flexDirection: 'column', gap: '4px' }}>
    {label && (
      <label style={{ fontSize: '0.875rem', fontWeight: 500, color: 'var(--intent-color-foreground)' }}>
        {label}
        {required && <span style={{ color: 'var(--intent-color-error)', marginLeft: '2px' }}>*</span>}
      </label>
    )}
    {children}
    {hint && !error && (
      <span style={{ fontSize: '0.75rem', color: 'var(--intent-color-muted)' }}>{hint}</span>
    )}
    {error && (
      <span style={{ fontSize: '0.75rem', color: 'var(--intent-color-error)' }}>{error}</span>
    )}
  </div>
)

export const Input = ({ type = 'text', placeholder, validation }) => (
  <input
    type={type}
    placeholder={placeholder}
    style={{
      padding: '8px 12px',
      border: `1px solid ${validation === 'invalid' ? 'var(--intent-color-error)' : 'var(--intent-color-border)'}`,
      borderRadius: '6px',
      fontSize: '0.875rem',
      width: '100%',
      outline: 'none'
    }}
  />
)

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `label` | `string` | - | Field label |
| `hint` | `string` | - | Helper text below input |
| `error` | `string` | - | Error message |
| `required` | `boolean` | `false` | Show required indicator |
| `children` | `ReactNode` | - | Input element |

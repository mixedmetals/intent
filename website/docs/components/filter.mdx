import React from 'react';

# Filter

Advanced filtering interface for data.

export const Filter = ({ fields, onChange }) => {
  const [values, setValues] = React.useState({})
  const [isOpen, setIsOpen] = React.useState(false)
  
  const handleChange = (key, value) => {
    const newValues = { ...values, [key]: value }
    setValues(newValues)
    onChange?.(newValues)
  }
  
  const clearFilters = () => {
    setValues({})
    onChange?.({})
  }
  
  const activeCount = Object.values(values).filter(v => v && v !== 'all').length
  
  return (
    <div style={{ position: 'relative' }}>
      <button 
        onClick={() => setIsOpen(!isOpen)}
        style={{
          display: 'flex',
          alignItems: 'center',
          gap: '8px',
          padding: '8px 16px',
          border: '1px solid var(--intent-color-border)',
          borderRadius: '8px',
          background: 'var(--intent-color-surface)',
          cursor: 'pointer',
          fontSize: '0.875rem'
        }}
      >
        <span>üîç</span>
        <span>Filter</span>
        {activeCount > 0 && (
          <span style={{ 
            padding: '2px 8px', 
            background: 'var(--intent-color-primary)', 
            color: 'white', 
            borderRadius: '9999px', 
            fontSize: '0.75rem' 
          }}>
            {activeCount}
          </span>
        )}
      </button>
      
      {isOpen && (
        <div style={{
          position: 'absolute',
          top: '100%',
          left: 0,
          marginTop: '8px',
          minWidth: '280px',
          background: 'var(--intent-color-surface)',
          border: '1px solid var(--intent-color-border)',
          borderRadius: '12px',
          boxShadow: '0 10px 15px -3px rgba(0,0,0,0.1)',
          zIndex: 50,
          padding: '16px'
        }}>
          <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginBottom: '16px' }}>
            <span style={{ fontWeight: 600 }}>Filters</span>
            {activeCount > 0 && (
              <button 
                onClick={clearFilters}
                style={{ fontSize: '0.875rem', color: 'var(--intent-color-primary)', background: 'none', border: 'none', cursor: 'pointer' }}
              >
                Clear all
              </button>
            )}
          </div>
          
          <div style={{ display: 'flex', flexDirection: 'column', gap: '12px' }}>
            {fields.map(field => (
              <div key={field.key}>
                <label style={{ display: 'block', fontSize: '0.875rem', fontWeight: 500, marginBottom: '4px' }}>{field.label}</label>
                {field.type === 'select' ? (
                  <select
                    value={values[field.key] || 'all'}
                    onChange={(e) => handleChange(field.key, e.target.value === 'all' ? '' : e.target.value)}
                    style={{
                      width: '100%',
                      padding: '8px 12px',
                      border: '1px solid var(--intent-color-border)',
                      borderRadius: '6px',
                      fontSize: '0.875rem'
                    }}
                  >
                    <option value="all">All</option>
                    {field.options.map(opt => (
                      <option key={opt} value={opt}>{opt}</option>
                    ))}
                  </select>
                ) : field.type === 'text' ? (
                  <input
                    type="text"
                    value={values[field.key] || ''}
                    onChange={(e) => handleChange(field.key, e.target.value)}
                    placeholder={`Search ${field.label.toLowerCase()}...`}
                    style={{
                      width: '100%',
                      padding: '8px 12px',
                      border: '1px solid var(--intent-color-border)',
                      borderRadius: '6px',
                      fontSize: '0.875rem'
                    }}
                  />
                ) : null}
              </div>
            ))}
          </div>
        </div>
      )}
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `fields` | `Array<{key: string, label: string, type: 'select' \| 'text' \| 'date', options?: string[]}>` | `[]` | Filter fields |
| `values` | `object` | `{}` | Current filter values |
| `onChange` | `(values: object) => void` | - | Filter change handler |

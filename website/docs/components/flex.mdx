# Flex

Flexible box container with alignment options.

export const Flex = ({ children, direction = 'row', gap = 'normal', justify, align, wrap, style }) => {
  const gapMap = { compact: '4px', normal: '8px', relaxed: '16px', loose: '24px' }
  return (
    <div style={{
      display: 'flex',
      flexDirection: direction,
      gap: gapMap[gap] || '8px',
      justifyContent: justify,
      alignItems: align,
      flexWrap: wrap ? 'wrap' : 'nowrap',
      ...style
    }}>
      {children}
    </div>
  )
}

export const Card = ({ children }) => (
  <div className="intent-card intent-card--padding-normal intent-card--elevation-low">{children}</div>
)

export const Button = ({ children, importance = 'secondary' }) => (
  <button className={`intent-button intent-button--importance-${importance} intent-button--size-medium`}>{children}</button>
)

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `direction` | `'row'` \| `'column'` | `'row'` | Flex direction |
| `gap` | `'compact'` \| `'normal'` \| `'relaxed'` \| `'loose'` | `'normal'` | Gap between items |
| `justify` | `'start'` \| `'center'` \| `'end'` \| `'between'` \| `'around'` | - | Justify content |
| `align` | `'start'` \| `'center'` \| `'end'` \| `'stretch'` | - | Align items |
| `wrap` | `boolean` | `false` | Wrap items |

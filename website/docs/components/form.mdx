import React from 'react';

# Form

Form building utilities with validation support.

:::tip
Use form components (Input, Select, Textarea, etc.) with your preferred form library.
:::

export const FormDemo = () => {
  const [values, setValues] = React.useState({ name: '', email: '', message: '' })
  const [errors, setErrors] = React.useState({})
  const [submitted, setSubmitted] = React.useState(false)
  
  const validate = () => {
    const newErrors = {}
    if (!values.name) newErrors.name = 'Name is required'
    if (!values.email) newErrors.email = 'Email is required'
    else if (!/\S+@\S+\.\S+/.test(values.email)) newErrors.email = 'Email is invalid'
    if (!values.message) newErrors.message = 'Message is required'
    setErrors(newErrors)
    return Object.keys(newErrors).length === 0
  }
  
  const handleSubmit = (e) => {
    e.preventDefault()
    if (validate()) {
      setSubmitted(true)
    }
  }
  
  if (submitted) {
    return (
      <div style={{ padding: '24px', textAlign: 'center', background: 'var(--intent-color-success-subtle)', borderRadius: '8px' }}>
        <div style={{ fontSize: '2rem', marginBottom: '8px' }}>âœ“</div>
        <div style={{ fontWeight: 600, color: 'var(--intent-color-success)' }}>Form submitted!</div>
      </div>
    )
  }
  
  return (
    <form onSubmit={handleSubmit} style={{ display: 'flex', flexDirection: 'column', gap: '16px' }}>
      <div>
        <label style={{ display: 'block', fontSize: '0.875rem', fontWeight: 500, marginBottom: '4px' }}>
          Name *
        </label>
        <input
          type="text"
          value={values.name}
          onChange={(e) => setValues({ ...values, name: e.target.value })}
          style={{
            width: '100%',
            padding: '8px 12px',
            border: `1px solid ${errors.name ? 'var(--intent-color-error)' : 'var(--intent-color-border)'}`,
            borderRadius: '6px',
            fontSize: '0.875rem'
          }}
          placeholder="Enter your name"
        />
        {errors.name && <div style={{ color: 'var(--intent-color-error)', fontSize: '0.75rem', marginTop: '4px' }}>{errors.name}</div>}
      </div>
      
      <div>
        <label style={{ display: 'block', fontSize: '0.875rem', fontWeight: 500, marginBottom: '4px' }}>
          Email *
        </label>
        <input
          type="email"
          value={values.email}
          onChange={(e) => setValues({ ...values, email: e.target.value })}
          style={{
            width: '100%',
            padding: '8px 12px',
            border: `1px solid ${errors.email ? 'var(--intent-color-error)' : 'var(--intent-color-border)'}`,
            borderRadius: '6px',
            fontSize: '0.875rem'
          }}
          placeholder="Enter your email"
        />
        {errors.email && <div style={{ color: 'var(--intent-color-error)', fontSize: '0.75rem', marginTop: '4px' }}>{errors.email}</div>}
      </div>
      
      <div>
        <label style={{ display: 'block', fontSize: '0.875rem', fontWeight: 500, marginBottom: '4px' }}>
          Message *
        </label>
        <textarea
          value={values.message}
          onChange={(e) => setValues({ ...values, message: e.target.value })}
          style={{
            width: '100%',
            padding: '8px 12px',
            border: `1px solid ${errors.message ? 'var(--intent-color-error)' : 'var(--intent-color-border)'}`,
            borderRadius: '6px',
            fontSize: '0.875rem',
            minHeight: '100px'
          }}
          placeholder="Enter your message"
        />
        {errors.message && <div style={{ color: 'var(--intent-color-error)', fontSize: '0.75rem', marginTop: '4px' }}>{errors.message}</div>}
      </div>
      
      <button type="submit" className="intent-button intent-button--importance-primary intent-button--size-medium">
        Submit
      </button>
    </form>
  )
}

## Form Libraries

Intent works well with popular form libraries:

### React Hook Form

```tsx
import { useForm } from 'react-hook-form';
import { Input, Button } from 'intent-react';

function MyForm() {
  const { register, handleSubmit } = useForm();
  
  return (
    <form onSubmit={handleSubmit(onSubmit)}>
      <Input label="Name" {...register('name')} />
      <Button type="submit">Submit</Button>
    </form>
  );
}
```

### Formik

```tsx
import { Formik, Form, Field } from 'formik';

<Formik initialValues={{ name: '' }} onSubmit={onSubmit}>
  <Form>
    <Field name="name" as={Input} label="Name" />
    <Button type="submit">Submit</Button>
  </Form>
</Formik>
```

## Design Notes

- Use consistent spacing between fields
- Show error states clearly
- Label all required fields
- Use appropriate input types
- Provide helpful placeholder text

import React from 'react';

# Form

Form building utilities with validation.

export const FormDemo = () => {
  const [values, setValues] = React.useState({ name: '', email: '', message: '' })
  const [errors, setErrors] = React.useState({})
  const [submitted, setSubmitted] = React.useState(false)
  
  const validate = () => {
    const newErrors = {}
    if (!values.name) newErrors.name = 'Name is required'
    if (!values.email) newErrors.email = 'Email is required'
    else if (!/\S+@\S+\.\S+/.test(values.email)) newErrors.email = 'Email is invalid'
    if (!values.message) newErrors.message = 'Message is required'
    setErrors(newErrors)
    return Object.keys(newErrors).length === 0
  }
  
  const handleSubmit = (e) => {
    e.preventDefault()
    if (validate()) {
      setSubmitted(true)
    }
  }
  
  if (submitted) {
    return (
      <div style={{ padding: '24px', textAlign: 'center', background: 'var(--intent-color-success-subtle, #dcfce7)', borderRadius: '8px' }}>
        <div style={{ fontSize: '2rem', marginBottom: '8px' }}>âœ“</div>
        <div style={{ fontWeight: 600, color: 'var(--intent-color-success, #16a34a)' }}>Form submitted!</div>
      </div>
    )
  }
  
  return (
    <form onSubmit={handleSubmit} style={{ display: 'flex', flexDirection: 'column', gap: '16px' }}>
      <div>
        <label style={{ display: 'block', fontSize: '0.875rem', fontWeight: 500, marginBottom: '4px' }}>
          Name *
        </label>
        <input
          type="text"
          value={values.name}
          onChange={(e) => setValues({ ...values, name: e.target.value })}
          style={{
            width: '100%',
            padding: '8px 12px',
            border: `1px solid ${errors.name ? 'var(--intent-color-error, #ef4444)' : 'var(--intent-color-border)'}`,
            borderRadius: '6px',
            fontSize: '0.875rem',
            outline: 'none'
          }}
          placeholder="Enter your name"
        />
        {errors.name && <div style={{ fontSize: '0.75rem', color: 'var(--intent-color-error, #ef4444)', marginTop: '4px' }}>{errors.name}</div>}
      </div>
      
      <div>
        <label style={{ display: 'block', fontSize: '0.875rem', fontWeight: 500, marginBottom: '4px' }}>
          Email *
        </label>
        <input
          type="email"
          value={values.email}
          onChange={(e) => setValues({ ...values, email: e.target.value })}
          style={{
            width: '100%',
            padding: '8px 12px',
            border: `1px solid ${errors.email ? 'var(--intent-color-error, #ef4444)' : 'var(--intent-color-border)'}`,
            borderRadius: '6px',
            fontSize: '0.875rem',
            outline: 'none'
          }}
          placeholder="Enter your email"
        />
        {errors.email && <div style={{ fontSize: '0.75rem', color: 'var(--intent-color-error, #ef4444)', marginTop: '4px' }}>{errors.email}</div>}
      </div>
      
      <div>
        <label style={{ display: 'block', fontSize: '0.875rem', fontWeight: 500, marginBottom: '4px' }}>
          Message *
        </label>
        <textarea
          value={values.message}
          onChange={(e) => setValues({ ...values, message: e.target.value })}
          rows={4}
          style={{
            width: '100%',
            padding: '8px 12px',
            border: `1px solid ${errors.message ? 'var(--intent-color-error, #ef4444)' : 'var(--intent-color-border)'}`,
            borderRadius: '6px',
            fontSize: '0.875rem',
            outline: 'none',
            resize: 'vertical',
            fontFamily: 'inherit'
          }}
          placeholder="Enter your message"
        />
        {errors.message && <div style={{ fontSize: '0.75rem', color: 'var(--intent-color-error, #ef4444)', marginTop: '4px' }}>{errors.message}</div>}
      </div>
      
      <button 
        type="submit"
        className="intent-btn intent-btn--variant-primary intent-btn--size-medium"
        style={{ alignSelf: 'flex-start' }}
      >
        Submit
      </button>
    </form>
  )
}

```tsx
<Form onSubmit={handleSubmit}>
  <Form.Field name="email">
    <Form.Label>Email</Form.Label>
    <Form.Input type="email" />
    <Form.Error />
  </Form.Field>
  <Form.Submit>Submit</Form.Submit>
</Form>
```

## Props

### Form
| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `onSubmit` | `(values) => void` | - | Submit handler |
| `validation` | `object` | - | Validation schema |
| `defaultValues` | `object` | `{}` | Default form values |
| `children` | `ReactNode` | - | Form fields |

### Form.Field
| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `name` | `string` | - | Field name |
| `children` | `ReactNode` | - | Field content |

### Form.Input
| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `type` | `string` | `'text'` | Input type |
| `placeholder` | `string` | - | Placeholder text |

# Grid

Two-dimensional layout system for complex arrangements.

export const Grid = ({ children, columns = 3, gap = 'normal', style }) => {
  const gapMap = { compact: '4px', normal: '8px', relaxed: '16px', loose: '24px' }
  return (
    <div style={{
      display: 'grid',
      gridTemplateColumns: `repeat(${columns}, 1fr)`,
      gap: gapMap[gap] || '8px',
      ...style
    }}>
      {children}
    </div>
  )
}

export const Card = ({ children, style }) => (
  <div className="intent-card intent-card--padding-normal intent-card--elevation-low" style={style}>
    {children}
  </div>
)

export const Text = ({ children }) => (
  <span className="intent-text">{children}</span>
)

  <Grid columns={3} gap="normal">


## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `columns` | `number` | `3` | Number of columns |
| `gap` | `'compact'` \| `'normal'` \| `'relaxed'` \| `'loose'` | `'normal'` | Grid gap |

## CSS Output

```css
.intent-grid {
  display: grid;
  grid-template-columns: repeat(3, 1fr);
  gap: var(--intent-spacing-normal);
}
```

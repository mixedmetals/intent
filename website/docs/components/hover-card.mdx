import React from 'react';

# HoverCard

Cards that appear on hover.

export const HoverCard = ({ trigger, children }) => {
  const [isVisible, setIsVisible] = React.useState(false)
  const [position, setPosition] = React.useState({ x: 0, y: 0 })
  
  const handleMouseEnter = (e) => {
    const rect = e.currentTarget.getBoundingClientRect()
    setPosition({ x: rect.left, y: rect.bottom + 8 })
    setIsVisible(true)
  }
  
  const handleMouseLeave = () => {
    setIsVisible(false)
  }
  
  return (
    <>
      <span 
        onMouseEnter={handleMouseEnter}
        onMouseLeave={handleMouseLeave}
        style={{ cursor: 'pointer', color: 'var(--intent-color-primary)', textDecoration: 'underline' }}
      >
        {trigger}
      </span>
      {isVisible && (
        <div 
          style={{ 
            position: 'fixed', 
            left: position.x, 
            top: position.y,
            maxWidth: '320px',
            background: 'var(--intent-color-surface)', 
            border: '1px solid var(--intent-color-border)',
            borderRadius: '8px',
            boxShadow: '0 10px 15px -3px rgba(0,0,0,0.1)',
            zIndex: 50,
            padding: '16px'
          }}
          onMouseEnter={() => setIsVisible(true)}
          onMouseLeave={handleMouseLeave}
        >
          {children}
        </div>
      )}
    </>
  )
}

```tsx
Hover over <HoverCard trigger="@username">
  <Avatar name="User" />
  <Text>User profile info...</Text>
</HoverCard> to see profile.
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `trigger` | `ReactNode` | - | Trigger element |
| `children` | `ReactNode` | - | Card content |
| `delay` | `number` | `200` | Delay before showing (ms) |

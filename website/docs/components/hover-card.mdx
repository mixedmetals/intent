import React from 'react';

# HoverCard

Cards that appear on hover. Uses the **Elevated** archetype.

export const HoverCardDemo = () => {
  const [isVisible, setIsVisible] = React.useState(false)
  
  return (
    <>
      <span 
        onMouseEnter={() => setIsVisible(true)}
        onMouseLeave={() => setIsVisible(false)}
        style={{ cursor: 'pointer', color: 'var(--intent-color-primary)', textDecoration: 'underline' }}
      >
        @intent
      </span>
      {isVisible && (
        <div 
          style={{ 
            position: 'fixed',
            left: '50%',
            top: '50%',
            transform: 'translate(-50%, -50%)',
            maxWidth: '320px',
            background: 'var(--intent-color-surface)', 
            border: '1px solid var(--intent-color-border)',
            borderRadius: '8px',
            boxShadow: '0 10px 15px -3px rgba(0,0,0,0.1)',
            zIndex: 50,
            padding: '16px'
          }}
        >
          <div style={{ display: 'flex', gap: '12px' }}>
            <div style={{ width: '40px', height: '40px', borderRadius: '50%', background: 'var(--intent-color-primary)', display: 'flex', alignItems: 'center', justifyContent: 'center', color: 'white', fontWeight: 'bold' }}>I</div>
            <div>
              <div style={{ fontWeight: 600 }}>Intent Framework</div>
              <div style={{ fontSize: '0.875rem', color: 'var(--intent-color-muted)' }}>Schema-first CSS framework</div>
            </div>
          </div>
        </div>
      )}
    </>
  )
}

## Examples

### Basic HoverCard

Hover over the trigger:

<div className="intent-demo">
  <div>
    Hover over <HoverCardDemo /> to see the card.
  </div>
</div>

```tsx
import { HoverCard } from 'intent-react';

<HoverCard trigger="@intent">
  <div>
    <Avatar src="..." />
    <div>
      <Text weight="semibold">Intent Framework</Text>
      <Text size="sm" color="muted">Schema-first CSS</Text>
    </div>
  </div>
</HoverCard>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `trigger` | `ReactNode` | - | Hover trigger element |
| `children` | `ReactNode` | - | Card content |
| `delay` | `number` | `200` | Show delay in ms |
| `side` | `'top' \| 'right' \| 'bottom' \| 'left'` | `'bottom'` | Card position |

## Design Notes

- Uses **Elevated** archetype (floating above content)
- Similar styling to Popover
- Delay prevents accidental triggers
- Supports rich content (avatars, text, etc.)

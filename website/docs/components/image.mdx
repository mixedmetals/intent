import React from 'react';

# Image

Responsive image component with fallback and loading skeleton. Uses the **Flat** archetype for the container.

export const ImageDemo = ({ src, alt, height = 200, rounded = true, objectFit = 'cover' }) => {
  const [loaded, setLoaded] = React.useState(false)
  const [error, setError] = React.useState(false)
  
  if (error) {
    return (
      <div style={{
        width: '100%',
        height: height || '200px',
        background: 'var(--intent-color-surface-elevated)',
        border: '1px solid var(--intent-color-border)',
        borderRadius: rounded ? '8px' : '0',
        display: 'flex',
        alignItems: 'center',
        justifyContent: 'center',
        color: 'var(--intent-color-muted)'
      }}>
        <div style={{ textAlign: 'center' }}>
          <div style={{ fontSize: '2rem', marginBottom: '8px' }}>üñºÔ∏è</div>
          <div>Failed to load</div>
        </div>
      </div>
    )
  }
  
  return (
    <div style={{ position: 'relative', width: '100%', height: height || 'auto' }}>
      {!loaded && (
        <div style={{
          position: 'absolute',
          inset: 0,
          background: 'var(--intent-color-surface-elevated)',
          borderRadius: rounded ? '8px' : '0',
          animation: 'pulse 2s cubic-bezier(0.4, 0, 0.6, 1) infinite'
        }} />
      )}
      <img
        src={src}
        alt={alt}
        onLoad={() => setLoaded(true)}
        onError={() => setError(true)}
        style={{
          width: '100%',
          height: height || '100%',
          objectFit,
          borderRadius: rounded ? '8px' : '0',
          opacity: loaded ? 1 : 0,
          transition: 'opacity 300ms'
        }}
      />
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `src` | `string` | - | Image source URL |
| `alt` | `string` | - | Alt text |
| `width` | `number \| string` | - | Image width |
| `height` | `number \| string` | - | Image height |
| `rounded` | `boolean` | `false` | Rounded corners |
| `objectFit` | `'cover' \| 'contain' \| 'fill' \| 'none'` | `'cover'` | Object fit style |
| `fallback` | `ReactNode` | - | Fallback content on error |

## Design Notes

- Uses **Flat** archetype (no bevel, no shadow)
- Loading state uses Skeleton animation
- Error fallback uses **Elevated** background surface
- Smooth opacity transition on load

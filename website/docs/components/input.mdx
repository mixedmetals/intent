import React from 'react';
import { Input, Stack } from 'intent-react';

# Input

Text input component with validation states. Uses the **Milled** archetype with hairline borders and inner shadows.

export const InputDemo = ({ placeholder, type = 'text', size = 'md', validation = 'none', disabled = false, defaultValue = '' }) => {
  const [value, setValue] = React.useState(defaultValue);
  
  return (
    <Input
      type={type}
      size={size}
      validation={validation}
      placeholder={placeholder}
      value={value}
      onChange={(e) => setValue(e.target.value)}
      disabled={disabled}
    />
  );
};

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `type` | `'text' \| 'password' \| 'email' \| 'number' \| ...` | `'text'` | Input type |
| `size` | `'xs' \| 'sm' \| 'md' \| 'lg' \| 'xl'` | `'md'` | Input size |
| `variant` | `'outline' \| 'filled' \| 'flushed' \| 'unstyled'` | `'outline'` | Visual variant |
| `validation` | `'none' \| 'valid' \| 'invalid' \| 'warning'` | `'none'` | Validation state |
| `placeholder` | `string` | - | Placeholder text |
| `value` | `string` | - | Controlled value |
| `defaultValue` | `string` | - | Uncontrolled default value |
| `disabled` | `boolean` | `false` | Disable input |
| `readOnly` | `boolean` | `false` | Read-only state |
| `loading` | `boolean` | `false` | Loading state |
| `fullWidth` | `boolean` | `false` | Full width |
| `onChange` | `(value: string) => void` | - | Change handler |

# Input

Text input component with validation states.

export const Input = ({ type = 'text', size = 'medium', validation = 'none', placeholder, value, readOnly, ...props }) => (
  <div className={`intent-input intent-input--size-${size} intent-input--validation-${validation}`}>
    <input 
      type={type}
      className="intent-input__field"
      placeholder={placeholder}
      value={value}
      readOnly={readOnly}
      {...props}
    />
  </div>
)

export const Stack = ({ children, gap = 'normal', style }) => (
  <div className={`intent-stack intent-stack--direction-column intent-stack--gap-${gap}`} style={style}>
    {children}
  </div>
)

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `type` | `'text'` \| `'email'` \| `'password'` \| `'number'` \| `'tel'` \| `'url'` | `'text'` | Input type |
| `size` | `'small'` \| `'medium'` \| `'large'` | `'medium'` | Input size |
| `validation` | `'valid'` \| `'invalid'` \| `'none'` | `'none'` | Validation state |
| `icon` | `string` | - | Left icon name |
| `disabled` | `boolean` | `false` | Disable input |
| `placeholder` | `string` | - | Placeholder text |

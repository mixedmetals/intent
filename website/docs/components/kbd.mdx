# Kbd

Keyboard key component for shortcuts.

export const Kbd = ({ children, size = 'md' }) => {
  const sizes = {
    sm: { padding: '2px 6px', fontSize: '0.75rem' },
    md: { padding: '4px 8px', fontSize: '0.875rem' },
    lg: { padding: '6px 12px', fontSize: '1rem' }
  }
  
  const s = sizes[size]
  
  return (
    <kbd style={{
      display: 'inline-flex',
      alignItems: 'center',
      justifyContent: 'center',
      padding: s.padding,
      fontSize: s.fontSize,
      fontFamily: 'ui-monospace, monospace',
      fontWeight: 600,
      color: 'var(--intent-color-foreground)',
      background: 'var(--intent-color-surface)',
      border: '1px solid var(--intent-color-border)',
      borderRadius: '6px',
      boxShadow: '0 2px 0 var(--intent-color-border)',
      whiteSpace: 'nowrap'
    }}>
      {children}
    </kbd>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | Key label |
| `size` | `'sm' \| 'md' \| 'lg'` | `'md'` | Key size |

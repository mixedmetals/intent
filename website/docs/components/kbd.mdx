# Kbd

Keyboard key indicator for shortcuts.

export const Kbd = ({ children, size = 'md' }) => {
  const padding = size === 'sm' ? '2px 6px' : '4px 10px'
  const fontSize = size === 'sm' ? '12px' : '14px'
  
  return (
    <kbd style={{
      display: 'inline-flex',
      alignItems: 'center',
      justifyContent: 'center',
      fontFamily: 'var(--intent-font-mono)',
      fontWeight: 500,
      fontSize,
      padding,
      border: '1px solid var(--intent-color-border)',
      borderRadius: '6px',
      background: 'var(--intent-color-surface-elevated)',
      boxShadow: '0 2px 0 var(--intent-color-border)',
      lineHeight: 1
    }}>
      {children}
    </kbd>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `string` | required | Key label |
| `size` | `'sm'` \| `'md'` | `'md'` | Key size |

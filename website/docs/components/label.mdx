# Label

Accessible form labels.

export const Label = ({ children, required, htmlFor, disabled }) => (
  <label 
    htmlFor={htmlFor}
    style={{
      display: 'inline-flex',
      alignItems: 'center',
      gap: '4px',
      fontSize: '0.875rem',
      fontWeight: 500,
      color: disabled ? 'var(--intent-color-muted)' : 'var(--intent-color-foreground)',
      cursor: disabled ? 'not-allowed' : 'pointer'
    }}
  >
    {children}
    {required && <span style={{ color: 'var(--intent-color-error, #ef4444)' }}>*</span>}
  </label>
)

## Examples

### Basic Label

<div className="intent-demo">
  <Label htmlFor="email">Email Address</Label>

      width: '100%',

</div>

<Label htmlFor="email">Email Address</Label>

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | Label text |
| `htmlFor` | `string` | - | Associated input ID |
| `required` | `boolean` | `false` | Show required indicator |
| `disabled` | `boolean` | `false` | Disabled state styling |

## CSS Output

```css
.intent-label {
  display: inline-flex;
  align-items: center;
  gap: 4px;
  font-size: 0.875rem;
  font-weight: 500;
  color: var(--intent-color-foreground);
  cursor: pointer;
}

.intent-label--required::after {
  content: "*";
  color: var(--intent-color-error);
}

.intent-label--disabled {
  color: var(--intent-color-muted);
  cursor: not-allowed;
}
```

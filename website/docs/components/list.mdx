import React from 'react';

# List

Display lists of items with various styles.

export const List = ({ children, variant = 'default' }) => (
  <ul style={{ 
    listStyle: 'none', 
    padding: 0, 
    margin: 0,
    border: variant === 'bordered' ? '1px solid var(--intent-color-border)' : 'none',
    borderRadius: variant === 'bordered' ? '8px' : '0',
    overflow: 'hidden'
  }}>
    {children}
  </ul>
)

export const ListItem = ({ children, icon, action }) => (
  <li style={{ 
    display: 'flex', 
    alignItems: 'center', 
    gap: '12px', 
    padding: '12px 16px',
    borderBottom: '1px solid var(--intent-color-border)'
  }}>
    {icon && <span>{icon}</span>}
    <span style={{ flex: 1 }}>{children}</span>
    {action}
  </li>
)

## Props

### List
| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `variant` | `'default' \| 'bordered'` | `'default'` | List style variant |
| `children` | `ReactNode` | - | ListItem components |

### ListItem
| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `icon` | `ReactNode` | - | Optional icon |
| `action` | `ReactNode` | - | Optional action element |
| `children` | `ReactNode` | - | Item content |

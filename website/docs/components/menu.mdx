import React from 'react';

# Menu

Dropdown menus for actions and navigation.

export const Menu = ({ children }) => {
  const [isOpen, setIsOpen] = React.useState(false)
  return (
    <div style={{ position: 'relative', display: 'inline-block' }}>
      <button 
        className="intent-btn intent-btn--variant-secondary intent-btn--size-medium"
        onClick={() => setIsOpen(!isOpen)}
      >
        Actions â–¼
      </button>
      {isOpen && (
        <div 
          style={{ 
            position: 'absolute', 
            top: '100%', 
            left: 0, 
            marginTop: '4px',
            minWidth: '200px',
            background: 'var(--intent-color-surface)', 
            border: '1px solid var(--intent-color-border)',
            borderRadius: '8px',
            boxShadow: '0 10px 15px -3px rgba(0,0,0,0.1)',
            zIndex: 50
          }}
        >
          <div onClick={() => setIsOpen(false)}>
            {children}
          </div>
        </div>
      )}
    </div>
  )
}

export const MenuItem = ({ children, icon, onClick }) => (
  <button
    onClick={onClick}
    style={{ 
      display: 'flex', 
      alignItems: 'center', 
      gap: '12px', 
      width: '100%',
      padding: '10px 16px',
      textAlign: 'left',
      background: 'none',
      border: 'none',
      cursor: 'pointer',
      fontSize: '0.875rem',
      transition: 'background 150ms'
    }}
    onMouseEnter={(e) => e.target.style.background = 'var(--intent-color-surface-hover)'}
    onMouseLeave={(e) => e.target.style.background = 'none'}
  >
    {icon && <span style={{ width: '16px', textAlign: 'center' }}>{icon}</span>}
    <span>{children}</span>
  </button>
)

export const MenuDivider = () => (
  <div style={{ height: '1px', background: 'var(--intent-color-border)', margin: '4px 0' }} />
)

## Props

### Menu
| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | Menu items |

### MenuItem
| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `icon` | `ReactNode` | - | Optional icon |
| `onClick` | `() => void` | - | Click handler |
| `children` | `ReactNode` | - | Item label |

### MenuDivider
| Prop | Type | Default | Description |
|------|------|---------|-------------|
| - | - | - | Visual separator |

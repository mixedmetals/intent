import React from 'react';

# Menubar

Apple-style menu bar for desktop applications. Uses the **Elevated** archetype.

export const MenubarDemo = () => {
  const [activeMenu, setActiveMenu] = React.useState(null)
  
  const items = [
    { label: 'File', children: [{ label: 'New' }, { label: 'Open' }, { separator: true }, { label: 'Exit' }] },
    { label: 'Edit', children: [{ label: 'Cut' }, { label: 'Copy' }, { label: 'Paste' }] },
    { label: 'View', children: [{ label: 'Zoom In' }, { label: 'Zoom Out' }] }
  ]
  
  return (
    <div 
      style={{ 
        display: 'flex', 
        gap: '4px', 
        padding: '4px 8px', 
        background: 'var(--intent-color-surface-elevated)',
        borderBottom: '1px solid var(--intent-color-border)'
      }}
      onMouseLeave={() => setActiveMenu(null)}
    >
      <div style={{ fontWeight: 600, padding: '4px 12px' }}>ðŸš€ Intent</div>
      {items.map((item, i) => (
        <div 
          key={i} 
          style={{ position: 'relative' }}
          onMouseEnter={() => setActiveMenu(i)}
        >
          <button 
            style={{ 
              padding: '4px 12px', 
              borderRadius: '4px', 
              border: 'none',
              background: activeMenu === i ? 'var(--intent-color-primary)' : 'transparent',
              color: activeMenu === i ? 'white' : 'var(--intent-color-foreground)',
              fontSize: '0.875rem',
              fontWeight: 500,
              cursor: 'pointer'
            }}
          >
            {item.label}
          </button>
          {activeMenu === i && item.children && (
            <div style={{
              position: 'absolute',
              top: '100%',
              left: 0,
              marginTop: '4px',
              minWidth: '200px',
              background: 'var(--intent-color-surface)',
              border: '1px solid var(--intent-color-border)',
              borderRadius: '8px',
              boxShadow: '0 10px 15px -3px rgba(0,0,0,0.1)',
              zIndex: 50,
              padding: '4px'
            }}>
              {item.children.map((child, j) => (
                <div key={j}>
                  {child.separator ? (
                    <div style={{ height: '1px', background: 'var(--intent-color-border)', margin: '4px' }} />
                  ) : (
                    <button
                      style={{
                        display: 'flex',
                        alignItems: 'center',
                        gap: '12px',
                        width: '100%',
                        padding: '6px 12px',
                        textAlign: 'left',
                        background: 'none',
                        border: 'none',
                        cursor: 'pointer',
                        fontSize: '0.875rem',
                        borderRadius: '4px'
                      }}
                      onMouseEnter={(e) => e.currentTarget.style.background = 'var(--intent-color-surface-hover)'}
                      onMouseLeave={(e) => e.currentTarget.style.background = 'none'}
                    >
                      {child.label}
                    </button>
                  )}
                </div>
              ))}
            </div>
          )}
        </div>
      ))}
    </div>
  )
}

## Examples

### Basic Menubar

<div className="intent-demo" style={{ padding: 0, overflow: 'hidden' }}>

```tsx
import { Menubar, MenubarMenu, MenubarItem } from 'intent-react';

<Menubar>
  <MenubarMenu label="File">
    <MenubarItem>New</MenubarItem>
    <MenubarItem>Open</MenubarItem>
    <MenubarSeparator />
    <MenubarItem>Exit</MenubarItem>
  </MenubarMenu>
</Menubar>
```

## Props

### Menubar

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | MenubarMenu components |

### MenubarMenu

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `label` | `string` | - | Menu label |
| `children` | `ReactNode` | - | Menu items |

### MenubarItem

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `shortcut` | `string` | - | Keyboard shortcut |
| `disabled` | `boolean` | `false` | Disabled state |
| `children` | `ReactNode` | - | Item label |

## Design Notes

- Uses **Elevated** archetype for background
- Hover state uses primary color
- Dropdowns use **Elevated** with shadow
- Keyboard navigation support
- Separator lines between groups

import React from 'react';

# Menubar

Apple-style menu bar for desktop applications.

export const Menubar = ({ items }) => {
  const [activeMenu, setActiveMenu] = React.useState(null)
  
  return (
    <div 
      style={{ 
        display: 'flex', 
        gap: '4px', 
        padding: '4px 8px', 
        background: 'var(--intent-color-surface-elevated)',
        borderBottom: '1px solid var(--intent-color-border)'
      }}
      onMouseLeave={() => setActiveMenu(null)}
    >
      {items.map((item, i) => (
        <div 
          key={i} 
          style={{ position: 'relative' }}
          onMouseEnter={() => setActiveMenu(i)}
        >
          <button 
            style={{ 
              padding: '4px 12px', 
              borderRadius: '4px', 
              border: 'none',
              background: activeMenu === i ? 'var(--intent-color-primary)' : 'transparent',
              color: activeMenu === i ? 'white' : 'var(--intent-color-foreground)',
              fontSize: '0.875rem',
              fontWeight: 500,
              cursor: 'pointer'
            }}
          >
            {item.label}
          </button>
          {activeMenu === i && item.children && (
            <div style={{
              position: 'absolute',
              top: '100%',
              left: 0,
              marginTop: '4px',
              minWidth: '200px',
              background: 'var(--intent-color-surface)',
              border: '1px solid var(--intent-color-border)',
              borderRadius: '8px',
              boxShadow: '0 10px 15px -3px rgba(0,0,0,0.1)',
              zIndex: 50,
              padding: '4px'
            }}>
              {item.children.map((child, j) => (
                <div key={j}>
                  {child.separator ? (
                    <div style={{ height: '1px', background: 'var(--intent-color-border)', margin: '4px' }} />
                  ) : (
                    <button
                      style={{
                        display: 'flex',
                        alignItems: 'center',
                        gap: '12px',
                        width: '100%',
                        padding: '6px 12px',
                        textAlign: 'left',
                        background: 'none',
                        border: 'none',
                        cursor: 'pointer',
                        fontSize: '0.875rem',
                        borderRadius: '4px'
                      }}
                      onMouseEnter={(e) => e.target.style.background = 'var(--intent-color-surface-hover)'}
                      onMouseLeave={(e) => e.target.style.background = 'none'}
                    >
                      <span>{child.label}</span>
                      {child.shortcut && <span style={{ marginLeft: 'auto', fontSize: '0.75rem', color: 'var(--intent-color-muted)' }}>{child.shortcut}</span>}
                    </button>
                  )}
                </div>
              ))}
            </div>
          )}
        </div>
      ))}
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `items` | `Array<{label: string, children?: Array<{label?: string, shortcut?: string, separator?: boolean}>}>` | `[]` | Menu items |

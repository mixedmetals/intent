import { Stack, Button, Modal, Text } from 'intent-react'

# Modal

Dialog window for focused interactions.

## Example

```tsx
import { useState } from 'react'
import { Button, Modal, Text, Stack } from 'intent-react'

function Example() {
  const [open, setOpen] = useState(false)
  
  return (
    <>
      <Button onClick={() => setOpen(true)}>
        Open Modal
      </Button>
      
      <Modal 
        open={open} 
        onOpenChange={setOpen}
        title="Example Modal"
      >
        <Text>Modal content goes here.</Text>
        <Stack direction="row" gap="normal">
          <Button importance="tertiary" onClick={() => setOpen(false)}>
            Cancel
          </Button>
          <Button importance="primary" onClick={() => setOpen(false)}>
            Confirm
          </Button>
        </Stack>
      </Modal>
    </>
  )
}
```

## Sizes

Modals support different widths:

| Size | Width | Use Case |
|------|-------|----------|
| `small` | 400px | Confirmations, simple forms |
| `medium` | 500px | Default, most content |
| `large` | 700px | Complex forms, detail views |
| `fullscreen` | 100% | Immersive experiences |

```tsx
<Modal size="large" open={open} onOpenChange={setOpen}>
  ...
</Modal>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `open` | `boolean` | - | Controlled open state |
| `onOpenChange` | `(open: boolean) => void` | - | State change callback |
| `title` | `string` | - | Modal header title |
| `size` | `'small'` \| `'medium'` \| `'large'` \| `'fullscreen'` | `'medium'` | Modal width |
| `showClose` | `boolean` | `true` | Show close button |
| `closeOnOverlay` | `boolean` | `true` | Close when clicking backdrop |

## Accessibility

- Traps focus within modal when open
- Returns focus to trigger on close
- Supports Escape key to close
- ARIA dialog role
- aria-modal="true"

## Pattern: Confirmation Dialog

```tsx
function DeleteConfirm({ item, onConfirm }) {
  const [open, setOpen] = useState(false)
  
  return (
    <>
      <Button importance="tertiary" onClick={() => setOpen(true)}>
        Delete
      </Button>
      
      <Modal
        open={open}
        onOpenChange={setOpen}
        title="Confirm Delete"
        size="small"
      >
        <Text>
          Are you sure you want to delete <strong>{item.name}</strong>?
        </Text>
        <Text size="sm" color="muted">
          This action cannot be undone.
        </Text>
        <Stack direction="row" gap="normal">
          <Button importance="tertiary" onClick={() => setOpen(false)}>
            Cancel
          </Button>
          <Button 
            importance="danger" 
            onClick={() => {
              onConfirm(item)
              setOpen(false)
            }}
          >
            Delete
          </Button>
        </Stack>
      </Modal>
    </>
  )
}
```

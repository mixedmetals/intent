import { Navbar, Button, Avatar } from 'intent-react'

# Navbar

Top navigation bar for application header.

## Example

```tsx
import { Navbar, Button, Avatar } from 'intent-react'

<Navbar
  logo={<Logo />}
  items={[
    { label: 'Home', href: '/', active: true },
    { label: 'Products', href: '/products' },
    { label: 'About', href: '/about' },
    { label: 'Contact', href: '/contact' },
  ]}
  actions={[
    <Button key="signin" importance="tertiary">Sign In</Button>,
    <Button key="signup" importance="primary">Sign Up</Button>,
  ]}
/>
```

## With Search

```tsx
<Navbar
  logo={<Logo />}
  items={[...]}
  search={
    <Input 
      icon="search" 
      placeholder="Search..."
      style={{ width: 240 }}
    />
  }
  actions={[...]}
/>
```

## Mobile Responsive

Navbar automatically collapses to a hamburger menu on mobile:

```tsx
<Navbar
  logo={<Logo />}
  mobileBreakpoint="md"
  items={[
    { label: 'Home', href: '/' },
    { label: 'Products', href: '/products' },
    { label: 'Documentation', href: '/docs' },
    { label: 'Pricing', href: '/pricing' },
  ]}
/>
```

## User Menu

```tsx
<Navbar
  logo={<Logo />}
  items={[...]}
  user={{
    name: 'John Doe',
    email: 'john@example.com',
    avatar: '/avatar.jpg',
    menuItems: [
      { label: 'Profile', href: '/profile' },

      { separator: true },
      { label: 'Sign Out', onClick: handleSignOut },
    ]

```

## Variants

### Default

Standard navbar with border:

```tsx
<Navbar variant="default" ... />
```

### Transparent

No background or border (for hero sections):

```tsx
<Navbar variant="transparent" ... />
```

### Sticky

Stays fixed at top when scrolling:

```tsx
<Navbar sticky ... />
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `logo` | `ReactNode` | - | Brand logo/element |
| `items` | `NavItem[]` | `[]` | Navigation items |
| `actions` | `ReactNode[]` | - | Right-side actions |
| `search` | `ReactNode` | - | Search input |
| `user` | `User` | - | User menu config |
| `variant` | `'default'` \| `'transparent'` | `'default'` | Navbar style |
| `sticky` | `boolean` | `false` | Stick to top |
| `mobileBreakpoint` | `'sm'` \| `'md'` \| `'lg'` | `'md'` | Mobile menu breakpoint |

### NavItem Type

```typescript
interface NavItem {
  label: string
  href?: string
  onClick?: () => void
  active?: boolean
  disabled?: boolean
  external?: boolean
  icon?: string
  items?: NavItem[]  // Dropdown items
}
```

### User Type

```typescript
interface User {
  name: string
  email?: string
  avatar?: string
  initials?: string
  menuItems: NavItem[]
}
```

## Simple Custom Navbar

If you need full control:

```tsx
<Navbar>
  <Navbar.Logo>
    <Logo />
  </Navbar.Logo>
  
  <Navbar.Items>
    <NavLink href="/">Home</NavLink>
    <NavLink href="/about">About</NavLink>
    <Dropdown trigger={<Button>More</Button>}>
      ...
    </Dropdown>
  </Navbar.Items>
  
  <Navbar.Actions>
    <Button importance="primary">Get Started</Button>
  </Navbar.Actions>
</Navbar>
```

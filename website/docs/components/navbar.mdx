# Navbar

Top navigation bar.

export const Navbar = ({ logo, items, actions }) => (
  <nav style={{ 
    display: 'flex', 
    alignItems: 'center', 
    justifyContent: 'space-between',
    padding: '12px 24px',
    background: 'var(--intent-color-surface)',
    borderBottom: '1px solid var(--intent-color-border)'
  }}>
    <div style={{ display: 'flex', alignItems: 'center', gap: '24px' }}>
      <div style={{ fontWeight: 700, fontSize: '1.25rem' }}>{logo}</div>
      <div style={{ display: 'flex', gap: '4px' }}>
        {items.map((item) => (
          <a
            key={item.label}
            href={item.href}
            style={{
              padding: '8px 12px',
              borderRadius: '6px',
              textDecoration: 'none',
              fontSize: '0.875rem',
              fontWeight: 500,
              color: item.active ? 'var(--intent-color-primary)' : 'var(--intent-color-foreground)',
              background: item.active ? 'var(--intent-color-surface-hover)' : 'transparent'
            }}
          >
            {item.label}
          </a>
        ))}
      </div>
    </div>
    <div style={{ display: 'flex', alignItems: 'center', gap: '12px' }}>
      {actions}
    </div>
  </nav>
)

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `logo` | `ReactNode` | - | Logo/content |
| `items` | `Array<{label: string, href: string, active?: boolean}>` | `[]` | Nav items |
| `actions` | `ReactNode` | - | Right side actions |

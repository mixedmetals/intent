import React from 'react';

# NavigationMenu

Top-level navigation with dropdowns.

export const NavigationMenu = ({ items }) => {
  const [activeMenu, setActiveMenu] = React.useState(null)
  
  return (
    <nav style={{ display: 'flex', gap: '4px' }}>
      {items.map((item, i) => (
        <div 
          key={i} 
          style={{ position: 'relative' }}
          onMouseEnter={() => item.children && setActiveMenu(i)}
          onMouseLeave={() => setActiveMenu(null)}
        >
          {item.children ? (
            <>
              <button 
                style={{ 
                  display: 'flex', 
                  alignItems: 'center', 
                  gap: '4px', 
                  padding: '8px 12px', 
                  borderRadius: '6px', 
                  border: 'none',
                  background: activeMenu === i ? 'var(--intent-color-primary)' : 'transparent',
                  color: activeMenu === i ? 'white' : 'var(--intent-color-foreground)',
                  cursor: 'pointer',
                  fontSize: '0.875rem',
                  fontWeight: 500
                }}
              >
                {item.label}
                <span style={{ fontSize: '0.75rem' }}>â–¼</span>
              </button>
              {activeMenu === i && (
                <div style={{
                  position: 'absolute',
                  top: '100%',
                  left: 0,
                  marginTop: '4px',
                  minWidth: '200px',
                  background: 'var(--intent-color-surface)',
                  border: '1px solid var(--intent-color-border)',
                  borderRadius: '8px',
                  boxShadow: '0 10px 15px -3px rgba(0,0,0,0.1)',
                  zIndex: 50,
                  padding: '8px'
                }}>
                  {item.children.map((child, j) => (
                    <a
                      key={j}
                      href={child.href}
                      style={{
                        display: 'block',
                        padding: '8px 12px',
                        borderRadius: '6px',
                        textDecoration: 'none',
                        color: 'var(--intent-color-foreground)',
                        fontSize: '0.875rem'
                      }}
                      onMouseEnter={(e) => e.currentTarget.style.background = 'var(--intent-color-surface-hover)'}
                      onMouseLeave={(e) => e.currentTarget.style.background = 'none'}
                    >
                      <div style={{ fontWeight: 500 }}>{child.label}</div>
                      {child.description && (
                        <div style={{ fontSize: '0.75rem', color: 'var(--intent-color-muted)', marginTop: '2px' }}>{child.description}</div>
                      )}
                    </a>
                  ))}
                </div>
              )}
            </>
          ) : (
            <a 
              href={item.href} 
              style={{ 
                display: 'flex', 
                alignItems: 'center', 
                gap: '4px', 
                padding: '8px 12px', 
                borderRadius: '6px', 
                textDecoration: 'none', 
                color: 'var(--intent-color-foreground)',
                fontSize: '0.875rem',
                fontWeight: 500
              }}
            >
              {item.label}
            </a>
          )}
        </div>
      ))}
    </nav>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `items` | `Array<{label: string, href?: string, children?: Array<{label: string, href: string, description?: string}>}>` | `[]` | Navigation items |

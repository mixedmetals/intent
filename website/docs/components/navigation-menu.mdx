import React from 'react';

# NavigationMenu

Top-level navigation with dropdowns. Uses the **Flat** archetype.

export const NavigationMenuDemo = () => {
  const [activeMenu, setActiveMenu] = React.useState(null)
  
  const items = [
    { 
      label: 'Products', 
      children: [
        { label: 'Overview', description: 'Learn about our products' },
        { label: 'Features', description: 'Explore capabilities' },
        { label: 'Pricing', description: 'View plans and pricing' }
      ] 
    },
    { 
      label: 'Resources', 
      children: [
        { label: 'Documentation' },
        { label: 'Tutorials' },
        { label: 'Blog' }
      ] 
    },
    { label: 'Pricing' }
  ]
  
  return (
    <nav style={{ display: 'flex', gap: '4px' }}>
      {items.map((item, i) => (
        <div 
          key={i} 
          style={{ position: 'relative' }}
          onMouseEnter={() => item.children && setActiveMenu(i)}
          onMouseLeave={() => setActiveMenu(null)}
        >
          {item.children ? (
            <>
              <button 
                style={{ 
                  display: 'flex', 
                  alignItems: 'center', 
                  gap: '4px', 
                  padding: '8px 12px', 
                  borderRadius: '6px', 
                  border: 'none',
                  background: activeMenu === i ? 'var(--intent-color-primary)' : 'transparent',
                  color: activeMenu === i ? 'white' : 'var(--intent-color-foreground)',
                  cursor: 'pointer',
                  fontSize: '0.875rem',
                  fontWeight: 500
                }}
              >
                {item.label}
                <span style={{ fontSize: '0.75rem' }}>â–¼</span>
              </button>
              {activeMenu === i && (
                <div style={{
                  position: 'absolute',
                  top: '100%',
                  left: 0,
                  marginTop: '4px',
                  minWidth: '200px',
                  background: 'var(--intent-color-surface)',
                  border: '1px solid var(--intent-color-border)',
                  borderRadius: '8px',
                  boxShadow: '0 10px 15px -3px rgba(0,0,0,0.1)',
                  zIndex: 50,
                  padding: '8px'
                }}>
                  {item.children.map((child, j) => (
                    <a
                      key={j}
                      href="#"
                      style={{
                        display: 'block',
                        padding: '8px 12px',
                        borderRadius: '6px',
                        textDecoration: 'none',
                        color: 'var(--intent-color-foreground)',
                        fontSize: '0.875rem'
                      }}
                      onMouseEnter={(e) => e.currentTarget.style.background = 'var(--intent-color-surface-hover)'}
                      onMouseLeave={(e) => e.currentTarget.style.background = 'none'}
                    >
                      <div style={{ fontWeight: 500 }}>{child.label}</div>
                      {child.description && (
                        <div style={{ fontSize: '0.75rem', color: 'var(--intent-color-muted)', marginTop: '2px' }}>{child.description}</div>
                      )}
                    </a>
                  ))}
                </div>
              )}
            </>
          ) : (
            <button 
              style={{ 
                padding: '8px 12px', 
                borderRadius: '6px', 
                border: 'none',
                background: 'transparent',
                color: 'var(--intent-color-foreground)',
                cursor: 'pointer',
                fontSize: '0.875rem',
                fontWeight: 500
              }}
            >
              {item.label}
            </button>
          )}
        </div>
      ))}
    </nav>
  )
}

```tsx
import { NavigationMenu, NavigationMenuItem } from 'intent-react';

<NavigationMenu>
  <NavigationMenuItem label="Products">
    <a href="/overview">Overview</a>
    <a href="/features">Features</a>
  </NavigationMenuItem>
  <NavigationMenuItem label="Pricing" href="/pricing" />
</NavigationMenu>
```

## Props

### NavigationMenu

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | Navigation items |

### NavigationMenuItem

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `label` | `string` | - | Item label |
| `href` | `string` | - | Link URL (if no children) |
| `children` | `ReactNode` | - | Dropdown content |

## Design Notes

- Uses **Flat** archetype
- Dropdown uses **Elevated** with shadow
- Hover state shows dropdown
- Supports links with descriptions
- Responsive on mobile

import React from 'react';

# Notification

Toast notifications and alerts.

export const Notification = ({ title, message, variant = 'info', onClose }) => {
  const variants = {
    info: { icon: 'ℹ️', background: '#eff6ff', border: '#bfdbfe', color: '#1e40af' },
    success: { icon: '✓', background: '#f0fdf4', border: '#bbf7d0', color: '#166534' },
    warning: { icon: '⚠️', background: '#fffbeb', border: '#fde68a', color: '#92400e' },
    error: { icon: '✕', background: '#fef2f2', border: '#fecaca', color: '#991b1b' }
  }
  
  const v = variants[variant]
  
  return (
    <div style={{ 
      display: 'flex', 
      alignItems: 'flex-start', 
      gap: '12px', 
      padding: '16px',
      background: v.background,
      border: `1px solid ${v.border}`,
      borderRadius: '8px',
      color: v.color
    }}>
      <span style={{ fontSize: '1.125rem' }}>{v.icon}</span>
      <div style={{ flex: 1 }}>
        <div style={{ fontWeight: 600, marginBottom: '4px' }}>{title}</div>
        <div style={{ fontSize: '0.875rem', opacity: 0.9 }}>{message}</div>
      </div>
      {onClose && (
        <button 
          onClick={onClose}
          style={{ background: 'none', border: 'none', cursor: 'pointer', fontSize: '1.25rem', color: v.color }}
        >
          ×
        </button>
      )}
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `title` | `string` | - | Notification title |
| `message` | `string` | - | Notification message |
| `variant` | `'info' \| 'success' \| 'warning' \| 'error'` | `'info'` | Notification style |
| `onClose` | `() => void` | - | Close handler |

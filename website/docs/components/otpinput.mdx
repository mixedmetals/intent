# OTPInput

One-time password input with individual digit fields.

## Examples

### Basic OTPInput

```tsx
import { useState } from 'react'
import { OTPInput } from 'intent-react'

function Example() {
  const [otp, setOtp] = useState('')
  
  return (
    <OTPInput 
      length={6}
      value={otp}
      onChange={setOtp}
    />
  )
}
```

### With Verification

```tsx
const [otp, setOtp] = useState('')
const [verifying, setVerifying] = useState(false)
const [error, setError] = useState('')

<OTPInput 
  length={6}
  value={otp}
  onChange={(value) => {
    setOtp(value)
    setError('')
    if (value.length === 6) {
      verifyOTP(value)
    }
  }}
  error={!!error}
/>
```

### Numeric Only

```tsx
<OTPInput 
  length={4}
  type="numeric"
  value={otp}
  onChange={setOtp}
/>
```

### Secure Input

```tsx
<OTPInput 
  length={6}
  secure  // Masks input like password
  value={otp}
  onChange={setOtp}
/>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `length` | `number` | `6` | Number of digits |
| `value` | `string` | - | Current value |
| `onChange` | `(value: string) => void` | - | Change callback |
| `type` | `'text'` \| `'numeric'` | `'text'` | Input type |
| `secure` | `boolean` | `false` | Mask input |
| `disabled` | `boolean` | `false` | Disable input |
| `error` | `boolean` | `false` | Error state |
| `autoFocus` | `boolean` | `true` | Auto-focus first field |

import React from 'react';

# Pagination

Page navigation controls.

export const Pagination = ({ total, pageSize = 10, currentPage: initialPage = 1, onChange }) => {
  const [currentPage, setCurrentPage] = React.useState(initialPage)
  const totalPages = Math.ceil(total / pageSize)
  
  const goToPage = (page) => {
    if (page < 1 || page > totalPages) return
    setCurrentPage(page)
    onChange?.(page)
  }
  
  const pages = []
  for (let i = 1; i <= totalPages; i++) {
    if (i === 1 || i === totalPages || (i >= currentPage - 1 && i <= currentPage + 1)) {
      pages.push(i)
    } else if (pages[pages.length - 1] !== '...') {
      pages.push('...')
    }
  }
  
  return (
    <div style={{ display: 'flex', alignItems: 'center', gap: '4px' }}>
      <button 
        onClick={() => goToPage(currentPage - 1)}
        disabled={currentPage === 1}
        style={{
          padding: '8px 12px',
          border: '1px solid var(--intent-color-border)',
          borderRadius: '6px',
          background: 'var(--intent-color-surface)',
          cursor: currentPage === 1 ? 'not-allowed' : 'pointer',
          opacity: currentPage === 1 ? 0.5 : 1,
          fontSize: '0.875rem'
        }}
      >
        ← Prev
      </button>
      
      {pages.map((page, i) => (
        <button
          key={i}
          onClick={() => typeof page === 'number' && goToPage(page)}
          disabled={page === '...'}
          style={{
            minWidth: '36px',
            padding: '8px',
            border: '1px solid var(--intent-color-border)',
            borderRadius: '6px',
            background: page === currentPage ? 'var(--intent-color-primary)' : 'var(--intent-color-surface)',
            color: page === currentPage ? 'white' : 'var(--intent-color-foreground)',
            cursor: page === '...' ? 'default' : 'pointer',
            fontSize: '0.875rem',
            fontWeight: 500
          }}
        >
          {page}
        </button>
      ))}
      
      <button 
        onClick={() => goToPage(currentPage + 1)}
        disabled={currentPage === totalPages}
        style={{
          padding: '8px 12px',
          border: '1px solid var(--intent-color-border)',
          borderRadius: '6px',
          background: 'var(--intent-color-surface)',
          cursor: currentPage === totalPages ? 'not-allowed' : 'pointer',
          opacity: currentPage === totalPages ? 0.5 : 1,
          fontSize: '0.875rem'
        }}
      >
        Next →
      </button>
    </div>
  )
}

## Examples

### Basic Pagination

<div className="intent-demo">
  <Pagination total={100} pageSize={10} />
</div>

```tsx
<Pagination 
  total={100} 
  pageSize={10} 
  currentPage={1}
  onChange={(page) => console.log(page)}
/>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `total` | `number` | - | Total number of items |
| `pageSize` | `number` | `10` | Items per page |
| `currentPage` | `number` | `1` | Current page |
| `onChange` | `(page: number) => void` | - | Page change handler |
| `showSizeChanger` | `boolean` | `false` | Show page size selector |
| `showTotal` | `boolean` | `false` | Show total count |

import React from 'react';
import { Pagination, PaginationItem } from 'intent-react/components/navigation';

# Pagination

Page navigation controls. Uses the **Flat** archetype for page items.

export const PaginationDemo = () => {
  const [currentPage, setCurrentPage] = React.useState(1)
  const totalPages = 10
  
  const pages = []
  for (let i = 1; i <= totalPages; i++) {
    if (i === 1 || i === totalPages || (i >= currentPage - 1 && i <= currentPage + 1)) {
      pages.push(i)
    } else if (pages[pages.length - 1] !== '...') {
      pages.push('...')
    }
  }
  
  return (
    <div style={{ display: 'flex', alignItems: 'center', gap: '4px' }}>
      <button 
        onClick={() => setCurrentPage(p => Math.max(1, p - 1))}
        disabled={currentPage === 1}
        className="intent-button intent-button--importance-secondary intent-button--size-small"
      >
        ← Prev
      </button>
      
      {pages.map((page, i) => (
        <button
          key={i}
          onClick={() => typeof page === 'number' && setCurrentPage(page)}
          disabled={page === '...'}
          className={`intent-button intent-button--size-small ${page === currentPage ? 'intent-button--importance-primary' : 'intent-button--importance-secondary'}`}
          style={{ minWidth: '36px' }}
        >
          {page}
        </button>
      ))}
      
      <button 
        onClick={() => setCurrentPage(p => Math.min(totalPages, p + 1))}
        disabled={currentPage === totalPages}
        className="intent-button intent-button--importance-secondary intent-button--size-small"
      >
        Next →
      </button>
    </div>
  )
}

## Examples

### Basic Pagination

<div className="intent-demo">
  <PaginationDemo />
</div>

```tsx
import { Pagination, PaginationItem } from 'intent-react/components/navigation';

<Pagination count={10} page={page} onChange={setPage} />
```

### Sizes

```tsx
<Pagination size="sm" count={10} />
<Pagination size="md" count={10} />
<Pagination size="lg" count={10} />
```

### With Boundaries

```tsx
<Pagination count={20} boundaryCount={2} />
```

## Props

### Pagination

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `count` | `number` | - | Total number of pages |
| `page` | `number` | - | Current page (controlled) |
| `defaultPage` | `number` | `1` | Default page |
| `onChange` | `(page: number) => void` | - | Page change callback |
| `size` | `'sm' \| 'md' \| 'lg'` | `'md'` | Pagination size |
| `boundaryCount` | `number` | `1` | Number of boundary pages |
| `siblingCount` | `number` | `1` | Number of sibling pages |

### PaginationItem

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `page` | `number \| string` | - | Page number or '...' |
| `selected` | `boolean` | `false` | Selected state |
| `disabled` | `boolean` | `false` | Disabled state |
| `type` | `'page' \| 'previous' \| 'next' \| 'start-ellipsis' \| 'end-ellipsis'` | - | Item type |

## Design Notes

- Page items use **Flat** archetype
- Active page uses filled style
- Disabled state uses 0.5 opacity
- Ellipsis items are not interactive

import { PhoneInput } from 'intent-react'

# PhoneInput

Phone number input with country code selection.

## Examples

### Basic PhoneInput

```tsx
import { useState } from 'react'
import { PhoneInput } from 'intent-react'

function Example() {
  const [phone, setPhone] = useState('')
  
  return (
    <PhoneInput 
      value={phone}
      onChange={setPhone}
      defaultCountry="US"
      placeholder="(555) 123-4567"
    />
  )
}
```

### With Country Select

```tsx
<PhoneInput 
  value={phone}
  onChange={setPhone}
  countries={['US', 'CA', 'GB', 'AU', 'DE']}
  defaultCountry="US"
/>
```

### With Validation

```tsx
<PhoneInput 
  value={phone}
  onChange={setPhone}
  validation={isValid ? 'valid' : 'invalid'}
  errorMessage={!isValid ? 'Please enter a valid phone number' : undefined}
/>
```

### International Format

```tsx
<PhoneInput 
  value={phone}
  onChange={setPhone}
  format="international"  // +1 (555) 123-4567
/>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | - | Phone number value |
| `onChange` | `(value: string) => void` | - | Change callback |
| `defaultCountry` | `string` | `'US'` | Default country code |
| `countries` | `string[]` | - | Allowed countries |
| `format` | `'national'` \| `'international'` | `'national'` | Number format |
| `placeholder` | `string` | - | Input placeholder |
| `disabled` | `boolean` | `false` | Disable input |
| `validation` | `'valid'` \| `'invalid'` \| `'none'` | `'none'` | Validation state |

import React from 'react';
import { Button } from 'intent-react';

# Popover

Floating content panels. Uses the **Elevated** archetype.

export const PopoverDemo = () => {
  const [isOpen, setIsOpen] = React.useState(false)
  const ref = React.useRef(null)
  
  React.useEffect(() => {
    const handleClickOutside = (e) => {
      if (ref.current && !ref.current.contains(e.target)) {
        setIsOpen(false)
      }
    }
    document.addEventListener('mousedown', handleClickOutside)
    return () => document.removeEventListener('mousedown', handleClickOutside)
  }, [])
  
  return (
    <div ref={ref} style={{ position: 'relative', display: 'inline-block' }}>
      <Button importance="secondary" onClick={() => setIsOpen(!isOpen)}>Open Popover</Button>
      {isOpen && (
        <div style={{ 
          position: 'absolute', 
          top: '100%', 
          left: 0, 
          marginTop: '8px',
          minWidth: '280px',
          background: 'var(--intent-color-surface)', 
          border: '1px solid var(--intent-color-border)',
          borderRadius: '8px',
          boxShadow: '0 10px 15px -3px rgba(0,0,0,0.1)',
          zIndex: 50
        }}>
          <div style={{ padding: '12px 16px', borderBottom: '1px solid var(--intent-color-border)', fontWeight: 600 }}>
            Dimensions
          </div>
          <div style={{ padding: '16px' }}>
            <div style={{ display: 'flex', flexDirection: 'column', gap: '12px' }}>
              <div>
                <label style={{ display: 'block', fontSize: '0.75rem', marginBottom: '4px', fontWeight: 500 }}>Width</label>
                <input type="text" defaultValue="100%" style={{ width: '100%', padding: '6px 10px', border: '1px solid var(--intent-color-border)', borderRadius: '4px' }} />
              </div>
              <div>
                <label style={{ display: 'block', fontSize: '0.75rem', marginBottom: '4px', fontWeight: 500 }}>Height</label>
                <input type="text" defaultValue="auto" style={{ width: '100%', padding: '6px 10px', border: '1px solid var(--intent-color-border)', borderRadius: '4px' }} />
              </div>
              <div style={{ display: 'flex', gap: '8px', marginTop: '8px' }}>
                <Button size="sm" style={{ flex: 1 }}>Save</Button>
              </div>
            </div>
          </div>
        </div>
      )}
    </div>
  )
}

## Examples

### Basic Popover

<div className="intent-demo">
  <PopoverDemo />
</div>

```tsx
import { Popover, Button } from 'intent-react';

<Popover>
  <PopoverTrigger>
    <Button>Open Popover</Button>
  </PopoverTrigger>
  <PopoverContent>
    <div>Popover content here</div>
  </PopoverContent>
</Popover>
```

### With Title

```tsx
<Popover>
  <PopoverTrigger>
    <Button>Settings</Button>
  </PopoverTrigger>
  <PopoverContent title="Display Settings">
    <Stack gap="normal">
      <Switch label="Show grid" />
      <Switch label="Snap to grid" />
    </Stack>
  </PopoverContent>
</Popover>
```

## Props

### Popover

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `open` | `boolean` | - | Controlled open state |
| `defaultOpen` | `boolean` | `false` | Default open state |
| `onOpenChange` | `(open: boolean) => void` | - | Open state change callback |

### PopoverContent

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `title` | `string` | - | Popover title |
| `align` | `'start' \| 'center' \| 'end'` | `'center'` | Alignment |
| `side` | `'top' \| 'right' \| 'bottom' \| 'left'` | `'bottom'` | Preferred side |

## Design Notes

- Uses **Elevated** archetype (floating above content)
- Arrow points to trigger element
- Click outside to close
- Respects safe area boundaries
- Same shadow as DropdownMenu

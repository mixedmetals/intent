import React from 'react';

# Popover

Floating content panels.

export const Popover = ({ trigger, children, title }) => {
  const [isOpen, setIsOpen] = React.useState(false)
  const ref = React.useRef(null)
  
  React.useEffect(() => {
    const handleClickOutside = (e) => {
      if (ref.current && !ref.current.contains(e.target)) {
        setIsOpen(false)
      }
    }
    document.addEventListener('mousedown', handleClickOutside)
    return () => document.removeEventListener('mousedown', handleClickOutside)
  }, [])
  
  return (
    <div ref={ref} style={{ position: 'relative', display: 'inline-block' }}>
      <div onClick={() => setIsOpen(!isOpen)}>{trigger}</div>
      {isOpen && (
        <div style={{ 
          position: 'absolute', 
          top: '100%', 
          left: 0, 
          marginTop: '8px',
          minWidth: '280px',
          background: 'var(--intent-color-surface)', 
          border: '1px solid var(--intent-color-border)',
          borderRadius: '8px',
          boxShadow: '0 10px 15px -3px rgba(0,0,0,0.1)',
          zIndex: 50,
          padding: title ? '0' : '16px'
        }}>
          {title && (
            <div style={{ padding: '12px 16px', borderBottom: '1px solid var(--intent-color-border)', fontWeight: 600 }}>
              {title}
            </div>
          )}
          <div style={{ padding: title ? '16px' : '0' }}>
            {children}
          </div>
        </div>
      )}
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `trigger` | `ReactNode` | - | Trigger element |
| `children` | `ReactNode` | - | Popover content |
| `title` | `string` | - | Optional header title |
| `placement` | `'top' \| 'bottom' \| 'left' \| 'right'` | `'bottom'` | Position |

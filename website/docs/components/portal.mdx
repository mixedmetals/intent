import React from 'react';

# Portal

Render content outside the component tree.

export const Portal = ({ children }) => {
  // Portal implementation would use ReactDOM.createPortal
  // This is a simplified demo
  return (
    <div style={{ 
      position: 'fixed', 
      bottom: '20px', 
      right: '20px', 
      padding: '16px 24px',
      background: 'var(--intent-color-surface)',
      border: '1px solid var(--intent-color-border)',
      borderRadius: '8px',
      boxShadow: '0 10px 15px -3px rgba(0,0,0,0.1)',
      zIndex: 9999
    }}>
      <div style={{ fontWeight: 600, marginBottom: '4px' }}>Portal Content</div>
      <div style={{ fontSize: '0.875rem', color: 'var(--intent-color-muted)' }}>
        Rendered outside parent tree
      </div>
    </div>
  )
}

```tsx
// Portal renders children to document.body
<Portal>
  <Modal />
</Portal>

// Or to a specific container
<Portal container={document.getElementById('modal-root')}>
  <Tooltip />
</Portal>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | Content to portal |
| `container` | `Element` | `document.body` | Target container |

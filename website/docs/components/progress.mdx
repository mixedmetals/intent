# Progress

Visual indicator for completion percentage.

export const Progress = ({ value = 0, size = 'medium', label }) => {
  const height = size === 'small' ? 4 : size === 'large' ? 12 : 8
  return (
    <div style={{ width: '100%' }}>
      {label && (
        <div style={{ display: 'flex', justifyContent: 'space-between', marginBottom: '8px', fontSize: '14px' }}>
          <span>{label}</span>
          <span>{value}%</span>
        </div>
      )}
      <div style={{ width: '100%', height, background: 'var(--intent-color-border)', borderRadius: '9999px', overflow: 'hidden' }}>
        <div style={{ width: `${value}%`, height: '100%', background: 'var(--intent-color-primary)', borderRadius: '9999px', transition: 'width 150ms ease' }} />
      </div>
    </div>
  )
}

export const Stack = ({ children, gap = 'normal' }) => (
  <div className={`intent-stack intent-stack--direction-column intent-stack--gap-${gap}`}>
    {children}
  </div>
)

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `number` | `0` | Current progress (0-100) |
| `size` | `'small'` \| `'medium'` \| `'large'` | `'medium'` | Progress height |
| `label` | `string` | - | Optional label text |
| `variant` | `'default'` \| `'success'` \| `'warning'` \| `'danger'` | `'default'` | Color variant |

import React from 'react';

# Radio

Single-selection from a group of options. Uses the **Milled** archetype with circular hairline borders.

export const RadioDemo = ({ label, name = 'demo', value, defaultChecked = false, disabled = false, size = 'md' }) => {
  const [selected, setSelected] = React.useState(defaultChecked ? value : '');
  
  return (
    <Radio
      name={name}
      value={value}
      label={label}
      checked={selected === value}
      onChange={() => setSelected(value)}
      disabled={disabled}
      size={size}
    />
  );
};

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `size` | `'sm' \| 'md' \| 'lg'` | `'md'` | Radio size |
| `variant` | `'default' \| 'filled' \| 'outline'` | `'default'` | Visual variant |
| `name` | `string` | - | Group name (required) |
| `value` | `string` | - | Radio value (required) |
| `checked` | `boolean` | - | Controlled checked state |
| `defaultChecked` | `boolean` | - | Uncontrolled default state |
| `disabled` | `boolean` | `false` | Disable interaction |
| `label` | `string` | - | Label text |
| `onChange` | `(checked: boolean) => void` | - | Change handler |

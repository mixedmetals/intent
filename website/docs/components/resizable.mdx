import React from 'react';

# Resizable

Resizable panel layouts.

export const Resizable = ({ children, defaultSize = 50, direction = 'horizontal' }) => {
  const [size, setSize] = React.useState(defaultSize)
  const [isDragging, setIsDragging] = React.useState(false)
  const containerRef = React.useRef(null)
  
  const handleMouseDown = () => setIsDragging(true)
  
  const handleMouseMove = React.useCallback((e) => {
    if (!isDragging || !containerRef.current) return
    const rect = containerRef.current.getBoundingClientRect()
    const newSize = direction === 'horizontal' 
      ? ((e.clientX - rect.left) / rect.width) * 100
      : ((e.clientY - rect.top) / rect.height) * 100
    setSize(Math.max(10, Math.min(90, newSize)))
  }, [isDragging, direction])
  
  const handleMouseUp = () => setIsDragging(false)
  
  React.useEffect(() => {
    if (isDragging) {
      document.addEventListener('mousemove', handleMouseMove)
      document.addEventListener('mouseup', handleMouseUp)
      return () => {
        document.removeEventListener('mousemove', handleMouseMove)
        document.removeEventListener('mouseup', handleMouseUp)
      }
    }
  }, [isDragging, handleMouseMove])
  
  return (
    <div 
      ref={containerRef}
      style={{ 
        display: 'flex', 
        flexDirection: direction === 'horizontal' ? 'row' : 'column',
        height: '200px',
        border: '1px solid var(--intent-color-border)',
        borderRadius: '8px',
        overflow: 'hidden'
      }}
    >
      <div style={{ 
        flex: `0 0 ${size}%`, 
        padding: '16px',
        background: 'var(--intent-color-surface)',
        overflow: 'auto'
      }}>
        Panel A ({Math.round(size)}%)
      </div>
      <div 
        onMouseDown={handleMouseDown}
        style={{ 
          flex: '0 0 4px',
          background: isDragging ? 'var(--intent-color-primary)' : 'var(--intent-color-border)',
          cursor: direction === 'horizontal' ? 'col-resize' : 'row-resize',
          transition: 'background 150ms'
        }}
      />
      <div style={{ 
        flex: `0 0 ${100 - size - 0.5}%`, 
        padding: '16px',
        background: 'var(--intent-color-surface-elevated)',
        overflow: 'auto'
      }}>
        Panel B ({Math.round(100 - size - 0.5)}%)
      </div>
    </div>
  )
}

## Examples

### Horizontal Split

<div className="intent-demo">
  <Resizable direction="horizontal" defaultSize={50} />
</div>

```tsx
<Resizable direction="horizontal" defaultSize={50}>
  <Resizable.Panel>Sidebar</Resizable.Panel>
  <Resizable.Handle />
  <Resizable.Panel>Main Content</Resizable.Panel>
</Resizable>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `direction` | `'horizontal' \| 'vertical'` | `'horizontal'` | Resize direction |
| `defaultSize` | `number` | `50` | Initial size percentage |
| `children` | `ReactNode` | - | Panel content |

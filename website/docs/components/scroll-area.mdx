import React from 'react';

# ScrollArea

Custom scrollable containers with styled scrollbars.

export const ScrollArea = ({ children, maxHeight = '200px' }) => (
  <div style={{ 
    maxHeight, 
    overflow: 'auto',
    border: '1px solid var(--intent-color-border)',
    borderRadius: '8px',
    padding: '12px'
  }}>
    {children}
  </div>
)

## Examples

### Vertical Scroll

<div className="intent-demo">
  <ScrollArea maxHeight="150px">
    <p>Scrollable content area with custom scrollbar styling.</p>
    <p style={{ marginTop: '100px' }}>More content below...</p>
    <p style={{ marginTop: '100px' }}>Even more content...</p>
    <p style={{ marginTop: '100px' }}>End of content.</p>
  </ScrollArea>
</div>

```tsx
import { ScrollArea } from 'intent-react';

<ScrollArea maxHeight="200px">
  <p>Long content...</p>
</ScrollArea>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `maxHeight` | `string` | `'200px'` | Maximum height |
| `maxWidth` | `string` | - | Maximum width |
| `orientation` | `'vertical' \| 'horizontal' \| 'both'` | `'vertical'` | Scroll direction |

## Design Notes

- Thin scrollbar (8px width)
- Subtle thumb color (border color)
- Hover state darkens thumb
- Smooth scrolling behavior
- Works in all modern browsers

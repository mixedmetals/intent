import { Search } from 'intent-react'

# Search

Search input with suggestions and filters.

## Examples

### Basic Search

```tsx
import { useState } from 'react'
import { Search } from 'intent-react'

function Example() {
  const [query, setQuery] = useState('')
  
  return (
    <Search 
      value={query}
      onChange={setQuery}
      placeholder="Search..."
      onSearch={(q) => console.log('Search:', q)}
    />
  )
}
```

### With Suggestions

```tsx
<Search 
  value={query}
  onChange={setQuery}
  suggestions={[
    { id: '1', label: 'Recent: Project Alpha' },
    { id: '2', label: 'Recent: Dashboard' },
  ]}
  onSelectSuggestion={(s) => navigate(s.id)}
/>
```

### With Filters

```tsx
<Search 
  value={query}
  onChange={setQuery}
  filters={[
    { key: 'type', label: 'Type', options: ['All', 'Document', 'Image'] },
    { key: 'date', label: 'Date', options: ['Any', 'Today', 'Week'] },
  ]}
  onFilterChange={(filters) => applyFilters(filters)}
/>
```

### Loading State

```tsx
<Search 
  value={query}
  onChange={setQuery}
  loading={isSearching}
/>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | - | Search query |
| `onChange` | `(value: string) => void` | - | Change callback |
| `onSearch` | `(value: string) => void` | - | Search submit callback |
| `placeholder` | `string` | `'Search...'` | Placeholder text |
| `suggestions` | `Suggestion[]` | - | Search suggestions |
| `filters` | `Filter[]` | - | Search filters |
| `loading` | `boolean` | `false` | Loading state |
| `clearable` | `boolean` | `true` | Show clear button |
| `autoFocus` | `boolean` | `false` | Auto focus input |

### Suggestion Type

```typescript
interface Suggestion {
  id: string
  label: string
  icon?: string
}
```

### Filter Type

```typescript
interface Filter {
  key: string
  label: string
  options: string[]
}
```

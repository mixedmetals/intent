import React from 'react';

# Search

Search input with autocomplete. Uses the **Milled** archetype for the input.

export const SearchDemo = ({ placeholder = "Search..." }) => {
  const [query, setQuery] = React.useState('')
  const [isOpen, setIsOpen] = React.useState(false)
  
  const suggestions = [
    'Button component',
    'Card layout',
    'Modal dialog',
    'Form validation',
    'Toast notifications'
  ]
  
  const filtered = suggestions.filter(s => 
    s.toLowerCase().includes(query.toLowerCase())
  )
  
  return (
    <div style={{ position: 'relative', width: '100%', maxWidth: '400px' }}>
      <div style={{ position: 'relative' }}>
        <span style={{ 
          position: 'absolute', 
          left: '12px', 
          top: '50%', 
          transform: 'translateY(-50%)',
          color: 'var(--intent-color-muted)'
        }}>ğŸ”</span>
        <input
          type="text"
          value={query}
          onChange={(e) => setQuery(e.target.value)}
          onFocus={() => setIsOpen(true)}
          onBlur={() => setTimeout(() => setIsOpen(false), 150)}
          placeholder={placeholder}
          style={{
            width: '100%',
            padding: '10px 12px 10px 40px',
            border: '1px solid var(--intent-color-border)',
            borderRadius: '8px',
            fontSize: '0.875rem',
            outline: 'none',
            background: 'var(--intent-color-surface)'
          }}
        />
        <span style={{ 
          position: 'absolute', 
          right: '12px', 
          top: '50%', 
          transform: 'translateY(-50%)',
          fontSize: '0.75rem',
          color: 'var(--intent-color-muted)',
          padding: '2px 6px',
          background: 'var(--intent-color-surface-elevated)',
          borderRadius: '4px'
        }}>
          âŒ˜K
        </span>
      </div>
      
      {isOpen && filtered.length > 0 && (
        <div style={{
          position: 'absolute',
          top: '100%',
          left: 0,
          right: 0,
          marginTop: '8px',
          background: 'var(--intent-color-surface)',
          border: '1px solid var(--intent-color-border)',
          borderRadius: '8px',
          boxShadow: '0 10px 15px -3px rgba(0,0,0,0.1)',
          zIndex: 50
        }}>
          {filtered.slice(0, 5).map((suggestion, i) => (
            <button
              key={i}
              onClick={() => { setQuery(suggestion); setIsOpen(false); }}
              style={{
                display: 'flex',
                alignItems: 'center',
                gap: '8px',
                width: '100%',
                padding: '10px 12px',
                textAlign: 'left',
                background: 'none',
                border: 'none',
                cursor: 'pointer',
                fontSize: '0.875rem'
              }}
              onMouseEnter={(e) => e.currentTarget.style.background = 'var(--intent-color-surface-hover)'}
              onMouseLeave={(e) => e.currentTarget.style.background = 'none'}
            >
              <span>ğŸ”</span>
              <span>{suggestion}</span>
            </button>
          ))}
        </div>
      )}
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `placeholder` | `string` | `"Search..."` | Input placeholder |
| `value` | `string` | - | Controlled value |
| `suggestions` | `string[]` | `[]` | Autocomplete suggestions |
| `onSearch` | `(query: string) => void` | - | Search handler |
| `onSelect` | `(value: string) => void` | - | Selection handler |
| `shortcut` | `string` | - | Keyboard shortcut hint |

## Design Notes

- Input uses **Milled** archetype
- Search icon positioned absolutely on left
- Shortcut hint on right side
- Dropdown uses **Elevated** archetype
- Hover state on suggestions

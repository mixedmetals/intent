import React from 'react';

# Select

Dropdown selection input.

export const Select = ({ label, options, placeholder = "Select an option..." }) => {
  const [isOpen, setIsOpen] = React.useState(false)
  const [selected, setSelected] = React.useState(null)
  
  return (
    <div style={{ width: '100%', maxWidth: '300px' }}>
      {label && <label style={{ display: 'block', fontSize: '0.875rem', fontWeight: 500, marginBottom: '4px' }}>{label}</label>}
      <div style={{ position: 'relative' }}>
        <button
          onClick={() => setIsOpen(!isOpen)}
          style={{
            width: '100%',
            padding: '8px 36px 8px 12px',
            border: '1px solid var(--intent-color-border)',
            borderRadius: '8px',
            background: 'var(--intent-color-surface)',
            textAlign: 'left',
            fontSize: '0.875rem',
            cursor: 'pointer',
            color: selected ? 'var(--intent-color-foreground)' : 'var(--intent-color-muted)'
          }}
        >
          {selected ? selected.label : placeholder}
        </button>
        <span style={{ 
          position: 'absolute', 
          right: '12px', 
          top: '50%', 
          transform: 'translateY(-50%)',
          pointerEvents: 'none',
          fontSize: '0.75rem'
        }}>▼</span>
        
        {isOpen && (
          <div style={{
            position: 'absolute',
            top: '100%',
            left: 0,
            right: 0,
            marginTop: '4px',
            background: 'var(--intent-color-surface)',
            border: '1px solid var(--intent-color-border)',
            borderRadius: '8px',
            boxShadow: '0 10px 15px -3px rgba(0,0,0,0.1)',
            zIndex: 50,
            maxHeight: '200px',
            overflow: 'auto'
          }}>
            {options.map((option) => (
              <button
                key={option.value}
                onClick={() => {
                  setSelected(option)
                  setIsOpen(false)
                }}
                style={{
                  width: '100%',
                  padding: '10px 12px',
                  textAlign: 'left',
                  background: selected?.value === option.value ? 'var(--intent-color-surface-hover)' : 'transparent',
                  border: 'none',
                  cursor: 'pointer',
                  fontSize: '0.875rem',
                  display: 'flex',
                  alignItems: 'center',
                  gap: '8px'
                }}
                onMouseEnter={(e) => e.target.style.background = 'var(--intent-color-surface-hover)'}
                onMouseLeave={(e) => e.target.style.background = selected?.value === option.value ? 'var(--intent-color-surface-hover)' : 'transparent'}
              >
                {selected?.value === option.value && <span>✓</span>}
                <span>{option.label}</span>
              </button>
            ))}
          </div>
        )}
      </div>
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `label` | `string` | - | Select label |
| `value` | `string` | - | Selected value |
| `options` | `Array<{value: string, label: string}>` | `[]` | Available options |
| `placeholder` | `string` | `"Select..."` | Placeholder text |
| `disabled` | `boolean` | `false` | Disable select |
| `onChange` | `(value: string) => void` | - | Change handler |

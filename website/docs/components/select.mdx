import React from 'react';

# Select

Dropdown selection input.

export const Select = ({ label, options, placeholder = "Select an option..." }) => {
  const [isOpen, setIsOpen] = React.useState(false)
  const [selected, setSelected] = React.useState(null)
  
  return (
    <div className="intent-select-wrapper" style={{ width: '100%', maxWidth: '300px' }}>
      {label && <label style={{ display: 'block', fontSize: '13px', fontWeight: 500, marginBottom: '6px', color: 'oklch(75% 0 0)' }}>{label}</label>}
      <button
        className="intent-select"
        onClick={() => setIsOpen(!isOpen)}
      >
        <span style={{ opacity: selected ? 1 : 0.5 }}>
          {selected ? selected.label : placeholder}
        </span>
        <svg width="12" height="12" viewBox="0 0 12 12" fill="none" stroke="currentColor" strokeWidth="2" style={{ transition: 'transform 150ms', transform: isOpen ? 'rotate(180deg)' : 'rotate(0)' }}>
          <path d="M3 5l3 3 3-3" />
        </svg>
      </button>
      
      {isOpen && (
        <div className="intent-select__dropdown">
          {options.map((option) => (
            <button
              key={option.value}
              className={`intent-select__option ${selected?.value === option.value ? 'intent-select__option--selected' : ''}`}
              onClick={() => {
                setSelected(option)
                setIsOpen(false)
              }}
            >
              {selected?.value === option.value && <span>âœ“</span>}
              <span>{option.label}</span>
            </button>
          ))}
        </div>
      )}
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `label` | `string` | - | Select label |
| `value` | `string` | - | Selected value |
| `options` | `Array<{value: string, label: string}>` | `[]` | Available options |
| `placeholder` | `string` | `"Select..."` | Placeholder text |
| `disabled` | `boolean` | `false` | Disable select |
| `onChange` | `(value: string) => void` | - | Change handler |

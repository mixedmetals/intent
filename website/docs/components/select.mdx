# Select

Dropdown for selecting from a list of options.

export const Select = ({ options = [], size = 'medium', validation = 'none', disabled, placeholder }) => (
  <div className={`intent-input intent-input--size-${size} intent-input--validation-${validation}`}>
    <select 
      className="intent-input__field intent-select"
      disabled={disabled}
      style={{ cursor: 'pointer', paddingRight: '40px' }}
    >
      {placeholder && <option value="">{placeholder}</option>}
      {options.map((opt, i) => (
        <option key={i} value={typeof opt === 'string' ? opt : opt.value}>
          {typeof opt === 'string' ? opt : opt.label}
        </option>
      ))}
    </select>
  </div>
)

export const Stack = ({ children, gap = 'normal' }) => (
  <div className={`intent-stack intent-stack--direction-column intent-stack--gap-${gap}`}>
    {children}
  </div>
)

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `options` | `Array<{value: string, label: string}>` \| `string[]` | `[]` | Select options |
| `size` | `'small'` \| `'medium'` \| `'large'` | `'medium'` | Select size |
| `validation` | `'valid'` \| `'invalid'` \| `'none'` | `'none'` | Validation state |
| `placeholder` | `string` | - | Default placeholder |
| `disabled` | `boolean` | `false` | Disable select |

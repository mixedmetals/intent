import React from 'react';
import { Select, Stack } from 'intent-react';

# Select

Dropdown selection input. Uses the **Milled** archetype with hairline borders.

export const SelectDemo = ({ placeholder, size = 'md', variant = 'outline', validation = 'none', disabled = false }) => {
  const [value, setValue] = React.useState('');
  
  const options = [
    { value: 'apple', label: 'Apple' },
    { value: 'banana', label: 'Banana' },
    { value: 'cherry', label: 'Cherry' },
    { value: 'date', label: 'Date' },
  ];
  
  return (
    <Select
      options={options}
      placeholder={placeholder}
      size={size}
      variant={variant}
      validation={validation}
      disabled={disabled}
      value={value}
      onChange={(v) => setValue(v)}
    />
  );
};

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `options` | `SelectOption[]` | - | Array of options |
| `size` | `'xs' \| 'sm' \| 'md' \| 'lg' \| 'xl'` | `'md'` | Select size |
| `variant` | `'outline' \| 'filled' \| 'flushed'` | `'outline'` | Visual variant |
| `validation` | `'none' \| 'valid' \| 'invalid' \| 'warning'` | `'none'` | Validation state |
| `placeholder` | `string` | - | Placeholder text |
| `value` | `string` | - | Selected value |
| `disabled` | `boolean` | `false` | Disable select |
| `onChange` | `(value: string) => void` | - | Change handler |

## SelectOption

```ts
interface SelectOption {
  value: string;
  label: string;
  disabled?: boolean;
}
```

import React from 'react';

# Sheet

Side panels that slide in from any edge.

export const Sheet = ({ title, children, isOpen, onClose, side = 'right' }) => {
  if (!isOpen) return null
  
  const positions = {
    top: { top: 0, left: 0, right: 0, height: 'auto', maxHeight: '50vh' },
    bottom: { bottom: 0, left: 0, right: 0, height: 'auto', maxHeight: '50vh' },
    left: { top: 0, left: 0, bottom: 0, width: 'auto', maxWidth: '400px' },
    right: { top: 0, right: 0, bottom: 0, width: 'auto', maxWidth: '400px' }
  }
  
  return (
    <div style={{ position: 'fixed', inset: 0, zIndex: 50 }}>
      <div 
        style={{ position: 'absolute', inset: 0, background: 'rgba(0,0,0,0.5)', backdropFilter: 'blur(4px)' }} 
        onClick={onClose}
      />
      <div style={{ 
        position: 'absolute', 
        background: 'var(--intent-color-surface)', 
        boxShadow: '0 25px 50px -12px rgba(0,0,0,0.25)',
        display: 'flex',
        flexDirection: 'column',
        ...positions[side]
      }}>
        <div style={{ padding: '16px 24px', borderBottom: '1px solid var(--intent-color-border)', display: 'flex', justifyContent: 'space-between', alignItems: 'center' }}>
          <h3 style={{ margin: 0, fontSize: '1.125rem', fontWeight: 600 }}>{title}</h3>
          <button 
            onClick={onClose}
            style={{ background: 'none', border: 'none', fontSize: '1.5rem', cursor: 'pointer', color: 'var(--intent-color-muted)' }}
          >
            Ã—
          </button>
        </div>
        <div style={{ padding: '24px', flex: 1, overflow: 'auto' }}>
          {children}
        </div>
      </div>
    </div>
  )
}

export const SheetDemo = () => {
  const [isOpen, setIsOpen] = React.useState(false)
  const [side, setSide] = React.useState('right')
  
  return (
    <div>
      <div style={{ display: 'flex', gap: '8px', flexWrap: 'wrap' }}>
        {['top', 'bottom', 'left', 'right'].map((s) => (
          <button 
            key={s}
            className="intent-btn intent-btn--variant-secondary intent-btn--size-medium"
            onClick={() => { setSide(s); setIsOpen(true); }}
          >
            Open {s}
          </button>
        ))}
      </div>
      <Sheet 
        isOpen={isOpen} 
        onClose={() => setIsOpen(false)}
        title={`${side.charAt(0).toUpperCase() + side.slice(1)} Sheet`}
        side={side}
      >
        <p>Sheet content goes here.</p>
      </Sheet>
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `isOpen` | `boolean` | `false` | Whether sheet is open |
| `onClose` | `() => void` | - | Close handler |
| `title` | `string` | - | Sheet title |
| `side` | `'top' \| 'bottom' \| 'left' \| 'right'` | `'right'` | Sheet position |
| `children` | `ReactNode` | - | Sheet content |

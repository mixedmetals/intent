import React from 'react';

# Skeleton

Loading placeholder for content that hasn't loaded yet.

export const Skeleton = ({ height = 20, width = '100%', circle = false }) => (
  <div
    style={{
      height,
      width,
      borderRadius: circle ? '50%' : '4px',
      background: 'linear-gradient(90deg, var(--intent-color-surface-elevated) 25%, var(--intent-color-surface) 50%, var(--intent-color-surface-elevated) 75%)',
      backgroundSize: '200% 100%',
      animation: 'skeleton-pulse 1.5s ease-in-out infinite'
    }}
  />
)

<style>{`
  @keyframes skeleton-pulse {
    0% { background-position: 200% 0; }
    100% { background-position: -200% 0; }
  }
`}</style>

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `height` | `number` | `20` | Skeleton height in px |
| `width` | `string \| number` | `'100%'` | Skeleton width |
| `circle` | `boolean` | `false` | Circular shape |

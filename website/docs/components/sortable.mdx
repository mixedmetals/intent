# Sortable

Drag and drop list reordering.

## Examples

### Basic Sortable

```tsx
import { useState } from 'react'
import { Sortable } from 'intent-react'

function Example() {
  const [items, setItems] = useState([
    { id: '1', name: 'Item 1' },
    { id: '2', name: 'Item 2' },
    { id: '3', name: 'Item 3' },
  ])
  
  return (
    <Sortable
      items={items}
      onReorder={setItems}
      renderItem={(item) => (
        <Card padding="normal">
          <Text>{item.name}</Text>
        </Card>
      )}
      getItemId={(item) => item.id}
    />
  )
}
```

### Horizontal Sortable

```tsx
<Sortable
  direction="horizontal"
  items={items}
  onReorder={setItems}
  renderItem={(item) => (
    <div style={{ width: 100, padding: 16, background: '#f0f0f0' }}>
      {item.name}
    </div>
  )}
/>
```

### With Handle

```tsx
<Sortable
  items={items}
  onReorder={setItems}
  renderItem={(item, { dragHandleProps }) => (
    <Card padding="normal">
      <div {...dragHandleProps} style={{ cursor: 'grab' }}>
        <Icon name="grip-vertical" />
      </div>
      <Text>{item.name}</Text>
    </Card>
  )}
/>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `items` | `T[]` | required | Sortable items |
| `onReorder` | `(items: T[]) => void` | required | Reorder callback |
| `renderItem` | `(item: T, props) => ReactNode` | required | Item renderer |
| `getItemId` | `(item: T) => string` | required | Get unique ID |
| `direction` | `'vertical'` \| `'horizontal'` | `'vertical'` | Sort direction |
| `disabled` | `boolean` | `false` | Disable sorting |

### Render Item Props

```typescript
interface RenderItemProps {
  dragHandleProps: object  // Props for drag handle
  isDragging: boolean      // Currently dragging
  isDragOver: boolean      // Drag over this item
}
```

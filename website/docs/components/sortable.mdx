import React from 'react';

# Sortable

Drag and drop sortable lists.

export const Sortable = ({ items: initialItems }) => {
  const [items, setItems] = React.useState(initialItems)
  const [dragging, setDragging] = React.useState(null)
  
  const handleDragStart = (index) => {
    setDragging(index)
  }
  
  const handleDragOver = (e, index) => {
    e.preventDefault()
    if (dragging === null || dragging === index) return
    
    const newItems = [...items]
    const item = newItems[dragging]
    newItems.splice(dragging, 1)
    newItems.splice(index, 0, item)
    setDragging(index)
    setItems(newItems)
  }
  
  const handleDragEnd = () => {
    setDragging(null)
  }
  
  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: '8px' }}>
      {items.map((item, index) => (
        <div
          key={item.id}
          draggable
          onDragStart={() => handleDragStart(index)}
          onDragOver={(e) => handleDragOver(e, index)}
          onDragEnd={handleDragEnd}
          style={{
            display: 'flex',
            alignItems: 'center',
            gap: '12px',
            padding: '12px 16px',
            background: 'var(--intent-color-surface)',
            border: '1px solid var(--intent-color-border)',
            borderRadius: '8px',
            cursor: 'grab',
            opacity: dragging === index ? 0.5 : 1,
            transition: 'all 150ms'
          }}
        >
          <span style={{ color: 'var(--intent-color-muted)', cursor: 'grab' }}>â‹®â‹®</span>
          <span style={{ fontSize: '1.25rem' }}>{item.icon}</span>
          <span style={{ flex: 1, fontWeight: 500 }}>{item.label}</span>
          <span style={{ fontSize: '0.75rem', color: 'var(--intent-color-muted)' }}>Drag to reorder</span>
        </div>
      ))}
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `items` | `Array<{id: string, label: string, icon?: ReactNode}>` | `[]` | Sortable items |
| `onReorder` | `(items) => void` | - | Reorder handler |
| `axis` | `'x' \| 'y'` | `'y'` | Drag axis |

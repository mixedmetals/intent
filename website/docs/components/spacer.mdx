# Spacer

Flexible spacing component for layouts.

export const Spacer = ({ size = 4 }) => {
  const sizes = {
    1: '4px',
    2: '8px',
    3: '12px',
    4: '16px',
    5: '20px',
    6: '24px',
    8: '32px',
    10: '40px',
    12: '48px',
    16: '64px'
  }
  
  return (
    <div style={{ 
      height: sizes[size] || `${size * 4}px`,
      background: 'repeating-linear-gradient(45deg, transparent, transparent 5px, var(--intent-color-border) 5px, var(--intent-color-border) 6px)',
      borderRadius: '4px',
      opacity: 0.3
    }} />
  )
}

## Examples

### Spacer Sizes

<div className="intent-demo">
  <div>Content above</div>
  <Spacer size={4} />
  <div>Content below (16px gap)</div>
  <Spacer size={8} />
  <div>More content (32px gap)</div>
</div>

```tsx
<div>Content A</div>
<Spacer size={4} />
<div>Content B</div>
```

### Horizontal Spacer

<div className="intent-demo">
  <div style={{ display: 'flex', alignItems: 'center' }}>
    <span>Item 1</span>
    <div style={{ width: '16px' }} />
    <span>Item 2</span>
    <div style={{ flex: 1 }} />
    <span>Item 3 (pushed right)</span>
  </div>
</div>

```tsx
<Flex>
  <Text>Left</Text>
  <Spacer flex={1} />
  <Text>Right</Text>
</Flex>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `size` | `number` | `4` | Spacer size (1-16) |
| `flex` | `number` | - | Flex grow value |

# Spinner

Loading indicator with multiple sizes.

export const Spinner = ({ size = 'medium', color = 'primary' }) => (
  <span className={`intent-spinner intent-spinner--size-${size} intent-spinner--color-${color}`} role="status" aria-label="Loading" />
)

export const Stack = ({ children, direction = 'row', gap = 'normal', align, style }) => (
  <div className={`intent-stack intent-stack--direction-${direction} intent-stack--gap-${gap} ${align ? `intent-stack--align-${align}` : ''}`} style={style}>
    {children}
  </div>
)

export const Button = ({ children, importance = 'secondary', size = 'medium', disabled, ...props }) => (
  <button className={`intent-button intent-button--importance-${importance} intent-button--size-${size}`} disabled={disabled} {...props}>
    {children}
  </button>
)

## Examples

### Sizes

<div className="intent-demo">
  <Stack direction="row" gap="normal" align="center">
    <Spinner size="small" />
    <Spinner size="medium" />
    <Spinner size="large" />
  </Stack>
</div>

```tsx
<Spinner size="small" />
<Spinner size="medium" />
<Spinner size="large" />
```

### Inside Button

<div className="intent-demo">
  <Stack direction="row" gap="normal">
    <Button disabled>
      <Spinner size="small" color="secondary" />
      <span style={{ marginLeft: 8 }}>Loading...</span>
    </Button>
  </Stack>
</div>

```tsx
<Button disabled>
  <Spinner size="small" color="secondary" />
  Loading...
</Button>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `size` | `'small'` \| `'medium'` \| `'large'` | `'medium'` | Spinner size |
| `color` | `'primary'` \| `'secondary'` \| `'muted'` | `'primary'` | Color variant |

## Accessibility

- Includes `aria-label="Loading"` by default
- Set `aria-busy="true"` on parent container
- Provide text alternative when visible

```tsx
<div aria-busy="true" aria-live="polite">
  <Spinner aria-label="Loading posts" />
  <span className="sr-only">Loading posts...</span>
</div>
```

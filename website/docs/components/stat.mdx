import React from 'react';

# Stat

Display statistics and metrics.

export const Stat = ({ label, value, helpText, trend }) => (
  <div style={{ padding: '16px', border: '1px solid var(--intent-color-border)', borderRadius: '8px', background: 'var(--intent-color-surface)' }}>
    <div style={{ fontSize: '0.875rem', color: 'var(--intent-color-muted)', marginBottom: '4px' }}>{label}</div>
    <div style={{ fontSize: '1.5rem', fontWeight: 600, marginBottom: '4px' }}>{value}</div>
    {helpText && <div style={{ fontSize: '0.75rem', color: 'var(--intent-color-muted)' }}>{helpText}</div>}
    {trend && (
      <div style={{ 
        fontSize: '0.875rem', 
        color: trend > 0 ? 'var(--intent-color-success)' : 'var(--intent-color-error)',
        marginTop: '4px',
        fontWeight: 500
      }}>
        {trend > 0 ? '↑' : '↓'} {Math.abs(trend)}%
      </div>
    )}
  </div>
)

export const StatGroup = ({ children }) => (
  <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fit, minmax(150px, 1fr))', gap: '16px' }}>
    {children}
  </div>
)

## Props

### Stat
| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `label` | `string` | - | Stat label |
| `value` | `string \| number` | - | Stat value |
| `helpText` | `string` | - | Help text below value |
| `trend` | `number` | - | Trend percentage (positive/negative) |

### StatGroup
| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | Stat components |

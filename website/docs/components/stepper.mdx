import React from 'react';

# Stepper

Multi-step process indicator. Uses the **Flat** archetype for step indicators.

export const StepperDemo = ({ steps, currentStep = 0 }) => (
  <div style={{ display: 'flex', alignItems: 'center', width: '100%' }}>
    {steps.map((step, i) => {
      const isActive = i === currentStep
      const isCompleted = i < currentStep
      
      return (
        <React.Fragment key={i}>
          <div style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', flex: i === steps.length - 1 ? '0 0 auto' : 1 }}>
            <div style={{
              width: '32px',
              height: '32px',
              borderRadius: '50%',
              display: 'flex',
              alignItems: 'center',
              justifyContent: 'center',
              fontSize: '0.875rem',
              fontWeight: 600,
              background: isCompleted ? 'var(--intent-color-success, #22c55e)' : isActive ? 'var(--intent-color-primary)' : 'var(--intent-color-surface-elevated)',
              color: isCompleted || isActive ? 'white' : 'var(--intent-color-muted)',
              border: `2px solid ${isCompleted ? 'var(--intent-color-success, #22c55e)' : isActive ? 'var(--intent-color-primary)' : 'var(--intent-color-border)'}`,
              transition: 'all 200ms'
            }}>
              {isCompleted ? 'âœ“' : i + 1}
            </div>
            <div style={{
              marginTop: '8px',
              fontSize: '0.75rem',
              fontWeight: 500,
              color: isActive ? 'var(--intent-color-primary)' : 'var(--intent-color-muted)',
              textAlign: 'center'
            }}>
              {step}
            </div>
          </div>
          {i < steps.length - 1 && (
            <div style={{
              flex: 1,
              height: '2px',
              background: isCompleted ? 'var(--intent-color-success, #22c55e)' : 'var(--intent-color-border)',
              margin: '0 8px',
              marginBottom: '20px'
            }} />
          )}
        </React.Fragment>
      )
    })}
  </div>
)

## Examples

### Horizontal Stepper

<div className="intent-demo">
  <StepperDemo 
    steps={['Account', 'Profile', 'Review', 'Complete']} 
    currentStep={1}
  />
</div>

```tsx
import { Stepper, Step } from 'intent-react';

<Stepper activeStep={1}>
  <Step label="Account" />
  <Step label="Profile" />
  <Step label="Review" />
  <Step label="Complete" />
</Stepper>
```

### Vertical Stepper

```tsx
<Stepper orientation="vertical" activeStep={1}>
  <Step label="Select campaign settings" description="Step 1" />
  <Step label="Create an ad group" description="Step 2" />
  <Step label="Create an ad" description="Step 3" />
</Stepper>
```

### With Error State

```tsx
<Stepper activeStep={1}>
  <Step label="Account" completed />
  <Step label="Profile" error />
  <Step label="Review" />
</Stepper>
```

## Props

### Stepper

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `activeStep` | `number` | `0` | Current active step index |
| `orientation` | `'horizontal' \| 'vertical'` | `'horizontal'` | Stepper direction |
| `alternativeLabel` | `boolean` | `false` | Labels below steps |

### Step

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `label` | `string` | - | Step label |
| `description` | `string` | - | Step description |
| `completed` | `boolean` | `false` | Completed state |
| `error` | `boolean` | `false` | Error state |

## Design Notes

- Step indicators use **Flat** archetype
- Active step uses primary color
- Completed steps use success color
- Connecting lines show progress
- Checkmark icon for completed steps

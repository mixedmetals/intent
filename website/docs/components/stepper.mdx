import { Stack, Stepper, Button } from 'intent-react'

# Stepper

Visual indicator for multi-step processes.

## Example

```tsx
import { useState } from 'react'
import { Stepper, Button, Stack } from 'intent-react'

function Checkout() {
  const [currentStep, setCurrentStep] = useState(1)
  
  const steps = [
    { id: 1, title: 'Shipping', description: 'Delivery address' },
    { id: 2, title: 'Payment', description: 'Billing method' },
    { id: 3, title: 'Review', description: 'Confirm order' },
  ]
  
  return (
    <Stack gap="relaxed">
      <Stepper steps={steps} currentStep={currentStep} />
      
      {/* Step content */}
      {currentStep === 1 && <ShippingForm />}
      {currentStep === 2 && <PaymentForm />}
      {currentStep === 3 && <ReviewOrder />}
      
      {/* Navigation */}
      <Stack direction="row" gap="normal">
        <Button 
          disabled={currentStep === 1}
          onClick={() => setCurrentStep(s => s - 1)}
        >
          Back
        </Button>
        <Button 
          importance="primary"
          onClick={() => setCurrentStep(s => s + 1)}
        >
          {currentStep === steps.length ? 'Complete' : 'Continue'}
        </Button>
      </Stack>
    </Stack>
  )
}
```

## Variants

### Horizontal (Default)

```tsx
<Stepper steps={steps} currentStep={2} />
```

### Vertical

```tsx
<Stepper variant="vertical" steps={steps} currentStep={2} />
```

### Simple (No Descriptions)

```tsx
<Stepper variant="simple" steps={steps} currentStep={2} />
```

## With Clickable Steps

```tsx
<Stepper 
  steps={steps}
  currentStep={currentStep}
  clickable
  onStepClick={(stepId) => {
    // Only allow clicking completed steps
    if (stepId < currentStep) {
      setCurrentStep(stepId)
    }
  }}
/>
```

## Error State

```tsx
<Stepper 
  steps={[
    { id: 1, title: 'Account', status: 'complete' },
    { id: 2, title: 'Personal', status: 'error', error: 'Invalid email' },
    { id: 3, title: 'Payment', status: 'pending' },
  ]}
  currentStep={2}
/>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `steps` | `Step[]` | required | Array of steps |
| `currentStep` | `number` | `1` | Active step ID |
| `variant` | `'horizontal'` \| `'vertical'` \| `'simple'` | `'horizontal'` | Stepper style |
| `clickable` | `boolean` | `false` | Allow step clicking |
| `onStepClick` | `(stepId: number) => void` | - | Step click handler |

### Step Type

```typescript
interface Step {
  id: number
  title: string
  description?: string
  status?: 'pending' | 'active' | 'complete' | 'error'
  error?: string
}
```

# Switch

Toggle switch for binary on/off states.

export const Switch = ({ label, description, checked, disabled }) => (
  <label style={{ display: 'flex', alignItems: 'flex-start', gap: '12px', cursor: disabled ? 'not-allowed' : 'pointer', opacity: disabled ? 0.5 : 1 }}>
    <div style={{ position: 'relative', display: 'inline-flex', alignItems: 'center' }}>
      <input 
        type="checkbox"
        style={{ position: 'absolute', opacity: 0, width: '100%', height: '100%', cursor: disabled ? 'not-allowed' : 'pointer' }}
        checked={checked}
        disabled={disabled}
      />
      <div style={{
        width: '44px',
        height: '24px',
        background: checked ? 'var(--intent-color-primary)' : 'var(--intent-color-border)',
        borderRadius: '9999px',
        position: 'relative',
        transition: 'background 150ms ease'
      }}>
        <div style={{
          position: 'absolute',
          top: '2px',
          left: checked ? '22px' : '2px',
          width: '20px',
          height: '20px',
          background: 'white',
          borderRadius: '50%',
          transition: 'transform 150ms ease',
          boxShadow: '0 1px 2px rgba(0,0,0,0.1)'
        }} />
      </div>
    </div>
    <div>
      <div style={{ fontWeight: 500 }}>{label}</div>
      {description && <div style={{ fontSize: '14px', color: 'var(--intent-color-text-muted)', marginTop: '2px' }}>{description}</div>}
    </div>
  </label>
)

export const Stack = ({ children, gap = 'normal' }) => (
  <div className={`intent-stack intent-stack--direction-column intent-stack--gap-${gap}`}>
    {children}
  </div>
)

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `label` | `string` | - | Switch label |
| `description` | `string` | - | Helper text |
| `checked` | `boolean` | - | Controlled state |
| `disabled` | `boolean` | `false` | Disable switch |
| `onChange` | `(checked: boolean) => void` | - | Change handler |

## When to Use

Use **Switch** for:
- Instant toggle actions (no submit required)
- Settings that take effect immediately
- Binary on/off preferences

Use **Checkbox** for:
- Form submissions
- Multiple selections
- Terms acceptance

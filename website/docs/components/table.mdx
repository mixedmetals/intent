import React from 'react';

# Table

Display tabular data with sorting and selection.

export const Table = ({ columns, data }) => (
  <table style={{ width: '100%', borderCollapse: 'collapse', border: '1px solid var(--intent-color-border)', borderRadius: '8px', overflow: 'hidden' }}>
    <thead>
      <tr style={{ background: 'var(--intent-color-surface-elevated)' }}>
        {columns.map((col) => (
          <th key={col.key} style={{ padding: '12px 16px', textAlign: 'left', fontWeight: 600, borderBottom: '1px solid var(--intent-color-border)' }}>
            {col.title}
          </th>
        ))}
      </tr>
    </thead>
    <tbody>
      {data.map((row, i) => (
        <tr key={i} style={{ borderBottom: '1px solid var(--intent-color-border)' }}>
          {columns.map((col) => (
            <td key={col.key} style={{ padding: '12px 16px' }}>
              {col.render ? col.render(row[col.key], row) : row[col.key]}
            </td>
          ))}
        </tr>
      ))}
    </tbody>
  </table>
)

export const Badge = ({ children, variant = 'default' }) => (
  <span className={`intent-badge intent-badge--variant-${variant} intent-badge--size-medium`}>{children}</span>
)

## Examples

### Basic Table

<div className="intent-demo">
  <Table 
    columns={[
      { key: 'name', title: 'Name' },
      { key: 'email', title: 'Email' },
      { key: 'role', title: 'Role' },
      { key: 'status', title: 'Status', render: (v) => <Badge variant={v === 'Active' ? 'success' : 'default'}>{v}</Badge> }
    ]}
    data={[
      { name: 'John Doe', email: 'john@example.com', role: 'Admin', status: 'Active' },
      { name: 'Jane Smith', email: 'jane@example.com', role: 'Editor', status: 'Active' },
      { name: 'Bob Johnson', email: 'bob@example.com', role: 'Viewer', status: 'Inactive' }
    ]}
  />
</div>

```tsx
<Table 
  columns={[
    { key: 'name', title: 'Name' },
    { key: 'email', title: 'Email' },
    { key: 'role', title: 'Role' }
  ]}
  data={[...]}
/>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `columns` | `Array<{key: string, title: string, render?: (value, row) => ReactNode}>` | `[]` | Column definitions |
| `data` | `Array<object>` | `[]` | Table data |

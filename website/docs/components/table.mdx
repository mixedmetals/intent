# Table

Display tabular data with sorting and selection.

## Example

<Table 
  columns={[
    { key: 'name', title: 'Name' },
    { key: 'email', title: 'Email' },
    { key: 'role', title: 'Role' },
    { key: 'status', title: 'Status', render: (value) => (
      <Badge variant={value === 'Active' ? 'success' : 'default'}>{value}</Badge>
    )}
  ]}
  data={[
    { id: 1, name: 'John Doe', email: 'john@example.com', role: 'Admin', status: 'Active' },
    { id: 2, name: 'Jane Smith', email: 'jane@example.com', role: 'Editor', status: 'Active' },
    { id: 3, name: 'Bob Johnson', email: 'bob@example.com', role: 'Viewer', status: 'Inactive' },
    { id: 4, name: 'Alice Brown', email: 'alice@example.com', role: 'Editor', status: 'Active' },
  ]}
  style={{ marginBottom: 24 }}
/>

## Sortable

```tsx
import { useState } from 'react'
import { Table } from 'intent-react'

function SortableTable() {
  const [sortKey, setSortKey] = useState('name')
  const [sortOrder, setSortOrder] = useState('asc')
  
  const handleSort = (key) => {
    if (sortKey === key) {
      setSortOrder(sortOrder === 'asc' ? 'desc' : 'asc')
    } else {
      setSortKey(key)
      setSortOrder('asc')
    }
  }
  
  const sortedData = [...data].sort((a, b) => {
    const aVal = a[sortKey]
    const bVal = b[sortKey]
    const order = sortOrder === 'asc' ? 1 : -1
    return aVal > bVal ? order : -order
  })
  
  return (
    <Table 
      sortable
      sortKey={sortKey}
      sortOrder={sortOrder}
      onSort={handleSort}
      columns={[...]}
      data={sortedData}
    />
  )
}
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `columns` | `Array<Column>` | `[]` | Column definitions |
| `data` | `Array<object>` | `[]` | Table data |
| `striped` | `boolean` | `false` | Alternating row colors |
| `bordered` | `boolean` | `false` | Border around table |
| `compact` | `boolean` | `false` | Reduced padding |
| `hover` | `boolean` | `false` | Row hover effect |
| `sortable` | `boolean` | `false` | Enable sorting |
| `sortKey` | `string` | - | Current sort column |
| `sortOrder` | `'asc'` \| `'desc'` | - | Current sort order |
| `onSort` | `(key: string) => void` | - | Sort callback |

### Column Type

```typescript
interface Column {
  key: string
  title: string
  width?: number | string
  align?: 'left' | 'center' | 'right'
  render?: (value: any, row: object) => ReactNode
}
```

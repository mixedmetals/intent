# Tabs

Organize content into tabbed sections. Uses compound component pattern.

export const TabsDemo = ({ variant = 'line' }) => {
  const [value, setValue] = React.useState(0);
  
  return (
    <Tabs value={value} onValueChange={setValue} variant={variant}>
      <TabList>
        <Tab>Account</Tab>
        <Tab>Password</Tab>
        <Tab>Notifications</Tab>
      </TabList>
      <TabPanels>
        <TabPanel>
          Manage your account settings and preferences.
        </TabPanel>
        <TabPanel>
          Change your password and security settings.
        </TabPanel>
        <TabPanel>
          Configure how you receive notifications.
        </TabPanel>
      </TabPanels>
    </Tabs>
  );
};

## Props

### Tabs

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `variant` | `'line' \| 'enclosed' \| 'enclosed-colored' \| 'soft-rounded' \| 'solid-rounded'` | `'line'` | Visual style |
| `size` | `'sm' \| 'md' \| 'lg'` | `'md'` | Tab size |
| `orientation` | `'horizontal' \| 'vertical'` | `'horizontal'` | Layout direction |
| `value` | `number` | - | Controlled selected index |
| `onValueChange` | `(value: number) => void` | - | Change handler |

### Tab

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `selected` | `boolean` | - | Selected state (internal) |
| `disabled` | `boolean` | `false` | Disable tab |

### TabPanel

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | Panel content |

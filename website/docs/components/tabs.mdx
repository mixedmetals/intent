import React from 'react';

# Tabs

Organize content into tabbed sections.

export const TabsDemo = () => {
  const [activeTab, setActiveTab] = React.useState('account')
  
  const tabs = [
    { id: 'account', label: 'Account', content: 'Manage your account settings and preferences.' },
    { id: 'password', label: 'Password', content: 'Change your password and security settings.' },
    { id: 'notifications', label: 'Notifications', content: 'Configure how you receive notifications.' }
  ]
  
  return (
    <div>
      <div style={{ display: 'flex', gap: '4px', borderBottom: '1px solid var(--intent-color-border)', marginBottom: '16px' }}>
        {tabs.map(tab => (
          <button
            key={tab.id}
            onClick={() => setActiveTab(tab.id)}
            style={{
              padding: '10px 16px',
              border: 'none',
              borderBottom: `2px solid ${activeTab === tab.id ? 'var(--intent-color-primary)' : 'transparent'}`,
              background: 'none',
              color: activeTab === tab.id ? 'var(--intent-color-primary)' : 'var(--intent-color-muted)',
              fontWeight: 500,
              fontSize: '0.875rem',
              cursor: 'pointer',
              marginBottom: '-1px'
            }}
          >
            {tab.label}
          </button>
        ))}
      </div>
      <div style={{ padding: '8px 0' }}>
        {tabs.find(t => t.id === activeTab)?.content}
      </div>
    </div>
  )
}

## Examples

### Basic Tabs

<div className="intent-demo">
  <TabsDemo />
</div>

```tsx
const [activeTab, setActiveTab] = useState('account')

<Tabs value={activeTab} onChange={setActiveTab}>
  <Tabs.List>
    <Tabs.Trigger value="account">Account</Tabs.Trigger>
    <Tabs.Trigger value="password">Password</Tabs.Trigger>
  </Tabs.List>
  <Tabs.Content value="account">
    <Text>Account settings...</Text>
  </Tabs.Content>
</Tabs>
```

## Props

### Tabs
| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `defaultValue` | `string` | - | Default active tab |
| `value` | `string` | - | Controlled active tab |
| `onChange` | `(value: string) => void` | - | Tab change handler |

### Tabs.Trigger
| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | required | Tab value |
| `children` | `ReactNode` | - | Tab label |

### Tabs.Content
| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | required | Associated tab value |
| `children` | `ReactNode` | - | Tab content |

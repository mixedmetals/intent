# Text

Typography component for consistent text styling.

export const Text = ({ children, size = 'base', weight, color = 'default', align, truncate }) => {
  const className = [
    'intent-text',
    `intent-text--size-${size}`,
    `intent-text--color-${color}`,
    weight && `intent-text--weight-${weight}`,
    align && `intent-text--align-${align}`,
    truncate && 'intent-text--truncate'
  ].filter(Boolean).join(' ')
  
  return <span className={className}>{children}</span>
}

export const Stack = ({ children, gap = 'normal' }) => (
  <div className={`intent-stack intent-stack--direction-column intent-stack--gap-${gap}`}>
    {children}
  </div>
)

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `size` | `'xs'` \| `'sm'` \| `'base'` \| `'lg'` \| `'xl'` \| `'2xl'` \| `'3xl'` \| `'4xl'` | `'base'` | Font size |
| `weight` | `'normal'` \| `'medium'` \| `'semibold'` \| `'bold'` | `'normal'` | Font weight |
| `color` | `'default'` \| `'muted'` \| `'primary'` \| `'success'` \| `'warning'` \| `'danger'` | `'default'` | Text color |
| `align` | `'left'` \| `'center'` \| `'right'` | `'left'` | Text alignment |
| `truncate` | `boolean` | `false` | Single line with ellipsis |
| `as` | `'p'` \| `'span'` \| `'div'` \| `'label'` | `'span'` | HTML element |

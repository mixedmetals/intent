# Textarea

Multi-line text input for longer content.

export const Textarea = ({ placeholder, rows = 3, size = 'medium', validation = 'none', disabled, readOnly }) => (
  <textarea
    className={`intent-input intent-textarea intent-input--size-${size} intent-input--validation-${validation}`}
    placeholder={placeholder}
    rows={rows}
    disabled={disabled}
    readOnly={readOnly}
    style={{ resize: 'vertical', minHeight: '80px', padding: '8px 12px', fontFamily: 'inherit' }}
  />
)

export const Stack = ({ children, gap = 'normal' }) => (
  <div className={`intent-stack intent-stack--direction-column intent-stack--gap-${gap}`}>
    {children}
  </div>
)

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `size` | `'small'` \| `'medium'` \| `'large'` | `'medium'` | Textarea size |
| `validation` | `'valid'` \| `'invalid'` \| `'none'` | `'none'` | Validation state |
| `rows` | `number` | `3` | Number of visible rows |
| `disabled` | `boolean` | `false` | Disable textarea |
| `placeholder` | `string` | - | Placeholder text |

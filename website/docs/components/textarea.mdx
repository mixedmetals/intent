import React from 'react';
import { Textarea, Stack } from 'intent-react';

# Textarea

Multi-line text input. Uses the **Milled** archetype with hairline borders.

export const TextareaDemo = ({ placeholder, rows = 4, disabled = false, readOnly = false, defaultValue = '' }) => {
  const [value, setValue] = React.useState(defaultValue);
  
  return (
    <Textarea
      placeholder={placeholder}
      rows={rows}
      disabled={disabled}
      readOnly={readOnly}
      value={value}
      onChange={(e) => setValue(e.target.value)}
    />
  );
};

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `placeholder` | `string` | - | Placeholder text |
| `rows` | `number` | `4` | Number of rows |
| `value` | `string` | - | Controlled value |
| `defaultValue` | `string` | - | Uncontrolled default |
| `disabled` | `boolean` | `false` | Disable textarea |
| `readOnly` | `boolean` | `false` | Read-only state |
| `onChange` | `(e: ChangeEvent) => void` | - | Change handler |

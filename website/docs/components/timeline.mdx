import React from 'react';

# Timeline

Display chronological events.

export const Timeline = ({ items }) => (
  <div style={{ position: 'relative', paddingLeft: '24px' }}>
    {items.map((item, i) => (
      <div key={i} style={{ position: 'relative', paddingBottom: '24px' }}>
        {/* Line */}
        <div style={{
          position: 'absolute',
          left: '6px',
          top: '8px',
          bottom: i === items.length - 1 ? '0' : '-24px',
          width: '2px',
          background: 'var(--intent-color-border)'
        }} />
        {/* Dot */}
        <div style={{
          position: 'absolute',
          left: '0',
          top: '4px',
          width: '12px',
          height: '12px',
          borderRadius: '50%',
          background: item.active ? 'var(--intent-color-primary)' : 'var(--intent-color-border)',
          border: '2px solid var(--intent-color-surface)'
        }} />
        {/* Content */}
        <div>
          <div style={{ fontSize: '0.875rem', color: 'var(--intent-color-muted)', marginBottom: '4px' }}>{item.time}</div>
          <div style={{ fontWeight: 500 }}>{item.title}</div>
          {item.description && <div style={{ fontSize: '0.875rem', color: 'var(--intent-color-muted)', marginTop: '4px' }}>{item.description}</div>}
        </div>
      </div>
    ))}
  </div>
)

## Examples

### Basic Timeline

<div className="intent-demo">
  <Timeline 
    items={[
      { time: 'Today, 2:00 PM', title: 'Pull Request Merged', description: 'feat: add new table component', active: true },
      { time: 'Today, 10:30 AM', title: 'Commit Pushed', description: 'fix: resolve sorting issue', active: false },
      { time: 'Yesterday, 4:15 PM', title: 'Issue Created', description: 'Bug: pagination not working', active: false }
    ]}
  />
</div>

```tsx
<Timeline 
  items={[
    { time: 'Today, 2:00 PM', title: 'PR Merged', description: 'feat: add component', active: true },
    { time: 'Today, 10:30 AM', title: 'Commit Pushed', active: false }
  ]}
/>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `items` | `Array<{time: string, title: string, description?: string, active?: boolean}>` | `[]` | Timeline items |

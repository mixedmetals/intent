import React from 'react';

# Toast

Temporary notification that appears and dismisses automatically.

export const Toast = ({ variant = 'info', title, children, onDismiss }) => {
  const variants = {
    info: { icon: 'ℹ️', bg: '#eff6ff', border: '#bfdbfe', color: '#1e40af' },
    success: { icon: '✓', bg: '#f0fdf4', border: '#bbf7d0', color: '#166534' },
    warning: { icon: '⚠️', bg: '#fffbeb', border: '#fde68a', color: '#92400e' },
    danger: { icon: '✕', bg: '#fef2f2', border: '#fecaca', color: '#991b1b' }
  }
  
  const v = variants[variant]
  
  React.useEffect(() => {
    if (onDismiss) {
      const timer = setTimeout(onDismiss, 5000)
      return () => clearTimeout(timer)
    }
  }, [onDismiss])
  
  return (
    <div style={{
      display: 'flex',
      gap: '12px',
      padding: '16px',
      background: v.bg,
      border: `1px solid ${v.border}`,
      borderRadius: '8px',
      color: v.color,
      maxWidth: '400px'
    }}>
      <span style={{ fontSize: '1.125rem', flexShrink: 0 }}>{v.icon}</span>
      <div style={{ flex: 1 }}>
        <div style={{ fontWeight: 600, marginBottom: '4px' }}>{title}</div>
        <div style={{ fontSize: '0.875rem' }}>{children}</div>
      </div>
      {onDismiss && (
        <button 
          onClick={onDismiss}
          style={{ background: 'none', border: 'none', cursor: 'pointer', color: v.color, fontSize: '1.25rem' }}
        >
          ×
        </button>
      )}
    </div>
  )
}

export const ToastDemo = () => {
  const [show, setShow] = React.useState(false)
  return (
    <div>
      <button 
        className="intent-btn intent-btn--variant-primary intent-btn--size-medium"
        onClick={() => setShow(true)}
      >
        Show Toast
      </button>
      {show && (
        <div style={{ position: 'fixed', bottom: '24px', right: '24px', zIndex: 50 }}>
          <Toast
            variant="success"
            title="Success!"
            onDismiss={() => setShow(false)}
          >
            Your changes have been saved.
          </Toast>
        </div>
      )}
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `variant` | `'info' \| 'success' \| 'warning' \| 'danger'` | `'info'` | Toast style |
| `title` | `string` | - | Toast title |
| `children` | `ReactNode` | - | Toast message |
| `onDismiss` | `() => void` | - | Dismiss callback (auto-dismisses after 5s) |

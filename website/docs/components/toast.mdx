import React from 'react';

# Toast

Temporary notification that appears and dismisses automatically. Uses the **Elevated** archetype.

export const Toast = (props) => {
  const variant = props.variant || 'info'
  const title = props.title
  const children = props.children
  const onDismiss = props.onDismiss
  
  return (
    <div style={{
      display: 'flex',
      alignItems: 'flex-start',
      gap: '12px',
      padding: '16px',
      background: 'var(--intent-color-surface)',
      border: '1px solid var(--intent-color-border)',
      borderRadius: '8px',
      boxShadow: '0 10px 15px -3px rgba(0,0,0,0.1)',
      minWidth: '300px'
    }}>
      <div style={{ flex: 1 }}>
        <div style={{ fontWeight: 600, marginBottom: '2px' }}>{title}</div>
        <div style={{ fontSize: '13px', opacity: 0.8 }}>{children}</div>
      </div>
      {onDismiss && (
        <button 
          onClick={onDismiss}
          style={{ background: 'none', border: 'none', cursor: 'pointer', fontSize: '1.25rem', color: 'var(--intent-color-muted)' }}
        >
          Ã—
        </button>
      )}
    </div>
  )
}

export const ToastDemo = () => {
  const [show, setShow] = React.useState(false)
  return (
    <div>
      <button
        className="intent-button intent-button--importance-primary intent-button--size-medium"
        onClick={() => setShow(true)}
      >
        Show Toast
      </button>
      {show && (
        <div style={{ position: 'fixed', bottom: '24px', right: '24px', zIndex: 9999 }}>
          <Toast
            title="Success!"
            onDismiss={() => setShow(false)}
          >
            Your changes have been saved.
          </Toast>
        </div>
      )}
    </div>
  )
}

## useToast Hook

```tsx
import { useToast } from 'intent-react';

function MyComponent() {
  const toast = useToast();
  
  return (
    <Button onClick={() => toast({ title: "Hello!" })}>
      Show Toast
    </Button>
  );
}
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `title` | `string` | - | Toast title |
| `description` | `string` | - | Toast message |
| `variant` | `'info' \| 'success' \| 'warning' \| 'error'` | `'info'` | Visual style |
| `duration` | `number` | `5000` | Auto-dismiss duration (ms) |
| `onDismiss` | `() => void` | - | Dismiss callback |

## Design Notes

- Uses **Elevated** archetype (floating above content)
- Fixed position in viewport corner
- Auto-dismisses after 5 seconds
- Progress bar shows remaining time
- Stacking support for multiple toasts

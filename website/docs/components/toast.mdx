import React from 'react';

# Toast

Temporary notification that appears and dismisses automatically.

export const Toast = ({ variant = 'info', title, children, onDismiss }) => {
  return (
    <div className={`intent-toast intent-toast--${variant}`}>
      <div style={{ flex: 1 }}>
        <div style={{ fontWeight: 600, marginBottom: '2px' }}>{title}</div>
        <div style={{ fontSize: '13px', opacity: 0.8 }}>{children}</div>
      </div>
      {onDismiss && (
        <button className="intent-toast__dismiss" onClick={onDismiss}>Ã—</button>
      )}
      {onDismiss && <div className="intent-toast__progress" />}
    </div>
  )
}

export const ToastDemo = () => {
  const [show, setShow] = React.useState(false)
  return (
    <div>
      <button
        className="intent-button intent-button--primary"
        onClick={() => setShow(true)}
      >
        Show Toast
      </button>
      {show && (
        <div style={{ position: 'fixed', bottom: '24px', right: '24px', zIndex: 9999 }}>
          <Toast
            variant="success"
            title="Success!"
            onDismiss={() => setShow(false)}
          >
            Your changes have been saved.
          </Toast>
        </div>
      )}
    </div>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `variant` | `'info' \| 'success' \| 'warning' \| 'danger'` | `'info'` | Toast style |
| `title` | `string` | - | Toast title |
| `children` | `ReactNode` | - | Toast message |
| `onDismiss` | `() => void` | - | Dismiss callback (auto-dismisses after 5s) |

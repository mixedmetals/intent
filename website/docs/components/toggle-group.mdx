import React from 'react';

# ToggleGroup

Group of toggle buttons. Uses the **Milled** archetype for buttons.

export const ToggleGroupDemo = ({ children, type = 'single', defaultValue }) => {
  const [value, setValue] = React.useState(type === 'single' ? defaultValue : [defaultValue].filter(Boolean))
  
  const items = [
    { value: 'left', label: 'Left' },
    { value: 'center', label: 'Center' },
    { value: 'right', label: 'Right' }
  ]
  
  return (
    <div style={{ display: 'inline-flex', gap: '4px', padding: '4px', border: '1px solid var(--intent-color-border)', borderRadius: '8px', background: 'var(--intent-color-surface)' }}>
      {items.map((item) => {
        const isActive = type === 'single' ? value === item.value : value.includes(item.value)
        return (
          <button
            key={item.value}
            onClick={() => {
              if (type === 'single') {
                setValue(item.value)
              } else {
                setValue(prev => 
                  prev.includes(item.value) 
                    ? prev.filter(v => v !== item.value)
                    : [...prev, item.value]
                )
              }
            }}
            style={{
              padding: '8px 12px',
              borderRadius: '6px',
              border: 'none',
              background: isActive ? 'var(--intent-color-primary)' : 'transparent',
              color: isActive ? 'white' : 'var(--intent-color-foreground)',
              fontSize: '0.875rem',
              fontWeight: 500,
              cursor: 'pointer',
              transition: 'all 150ms'
            }}
          >
            {item.label}
          </button>
        )
      })}
    </div>
  )
}

## Props

### ToggleGroup

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `type` | `'single' \| 'multiple'` | `'single'` | Selection mode |
| `value` | `string \| string[]` | - | Controlled value(s) |
| `defaultValue` | `string \| string[]` | - | Default value(s) |
| `onValueChange` | `(value) => void` | - | Change handler |

### ToggleGroupItem

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | - | Item value |
| `children` | `ReactNode` | - | Item content |

## Design Notes

- Uses **Milled** archetype for buttons (tactile feel)
- Container uses **Flat** archetype with border
- Active state uses filled primary style
- Gap between items for visual separation
- Smooth transition on state change

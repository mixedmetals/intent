import React from 'react';
import { Switch } from 'intent-react';

# Toggle

Two-state buttons that can be on or off. Uses the **Milled** archetype.

:::tip
For boolean toggles, use the `Switch` component. For button groups, use `ToggleGroup`.
:::

export const ToggleDemo = (props) => {
  const children = props.children
  const defaultPressed = props.defaultPressed || false
  const variant = props.variant || 'default'
  const disabled = props.disabled || false
  const [pressed, setPressed] = React.useState(defaultPressed)
  
  const variants = {
    default: {
      background: pressed ? 'var(--intent-color-surface-elevated)' : 'transparent',
      border: '1px solid var(--intent-color-border)'
    },
    outline: {
      background: pressed ? 'var(--intent-color-primary)' : 'transparent',
      color: pressed ? 'white' : 'var(--intent-color-foreground)',
      border: '1px solid ' + (pressed ? 'var(--intent-color-primary)' : 'var(--intent-color-border)')
    }
  }
  
  return (
    <button
      aria-pressed={pressed}
      onClick={() => setPressed(!pressed)}
      disabled={disabled}
      style={Object.assign({
        display: 'inline-flex',
        alignItems: 'center',
        justifyContent: 'center',
        gap: '8px',
        padding: '8px 12px',
        borderRadius: '6px',
        fontSize: '0.875rem',
        fontWeight: 500,
        cursor: disabled ? 'not-allowed' : 'pointer',
        opacity: disabled ? 0.5 : 1,
        transition: 'all 150ms'
      }, variants[variant])}
    >
      {children}
    </button>
  )
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `defaultPressed` | `boolean` | `false` | Initial pressed state |
| `pressed` | `boolean` | - | Controlled pressed state |
| `onPressedChange` | `(pressed: boolean) => void` | - | State change callback |
| `variant` | `'default' \| 'outline'` | `'default'` | Visual style |
| `disabled` | `boolean` | `false` | Disabled state |

## Design Notes

- Uses **Milled** archetype (slight bevel for tactile feel)
- `aria-pressed` attribute for accessibility
- Clear visual distinction between pressed/unpressed states
- Works as standalone button or in ToggleGroup

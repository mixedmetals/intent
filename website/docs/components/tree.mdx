import React from 'react';

# Tree

Hierarchical tree view component.

export const Tree = ({ items, defaultExpanded = [] }) => {
  const [expanded, setExpanded] = React.useState(new Set(defaultExpanded))
  
  const toggle = (id) => {
    setExpanded(prev => {
      const next = new Set(prev)
      if (next.has(id)) next.delete(id)
      else next.add(id)
      return next
    })
  }
  
  const renderItem = (item, level = 0) => {
    const hasChildren = item.children && item.children.length > 0
    const isExpanded = expanded.has(item.id)
    
    return (
      <div key={item.id}>
        <div 
          style={{ 
            display: 'flex', 
            alignItems: 'center', 
            gap: '4px', 
            padding: '6px 8px',
            paddingLeft: `${level * 20 + 8}px`,
            cursor: hasChildren ? 'pointer' : 'default',
            borderRadius: '4px'
          }}
          onClick={() => hasChildren && toggle(item.id)}
          onMouseEnter={(e) => e.target.style.background = 'var(--intent-color-surface-hover)'}
          onMouseLeave={(e) => e.target.style.background = 'transparent'}
        >
          {hasChildren && (
            <span style={{ 
              fontSize: '0.75rem', 
              width: '16px',
              transform: isExpanded ? 'rotate(90deg)' : 'rotate(0)',
              transition: 'transform 150ms'
            }}>â–¶</span>
          )}
          {!hasChildren && <span style={{ width: '16px' }} />}
          <span>{item.icon}</span>
          <span style={{ fontSize: '0.875rem' }}>{item.label}</span>
        </div>
        {hasChildren && isExpanded && (
          <div>
            {item.children.map(child => renderItem(child, level + 1))}
          </div>
        )}
      </div>
    )
  }
  
  return <div style={{ border: '1px solid var(--intent-color-border)', borderRadius: '8px', padding: '8px' }}>{items.map(item => renderItem(item))}</div>
}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `items` | `Array<{id: string, label: string, icon?: ReactNode, children?: array}>` | `[]` | Tree items |
| `defaultExpanded` | `string[]` | `[]` | Initially expanded item IDs |
| `onSelect` | `(id: string) => void` | - | Selection handler |

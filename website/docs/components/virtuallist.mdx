import React from 'react';

# VirtualList

Efficiently render large lists.

export const VirtualList = ({ items, itemHeight = 40, height = 200 }) => {
  const [scrollTop, setScrollTop] = React.useState(0)
  const containerRef = React.useRef(null)
  
  const visibleCount = Math.ceil(height / itemHeight)
  const totalHeight = items.length * itemHeight
  const startIndex = Math.floor(scrollTop / itemHeight)
  const endIndex = Math.min(startIndex + visibleCount + 1, items.length)
  
  const visibleItems = items.slice(startIndex, endIndex)
  const offsetY = startIndex * itemHeight
  
  return (
    <div
      ref={containerRef}
      onScroll={(e) => setScrollTop(e.target.scrollTop)}
      style={{
        height,
        overflow: 'auto',
        border: '1px solid var(--intent-color-border)',
        borderRadius: '8px'
      }}
    >
      <div style={{ height: totalHeight, position: 'relative' }}>
        <div style={{ transform: `translateY(${offsetY}px)` }}>
          {visibleItems.map((item, i) => (
            <div
              key={startIndex + i}
              style={{
                height: itemHeight,
                display: 'flex',
                alignItems: 'center',
                padding: '0 16px',
                borderBottom: '1px solid var(--intent-color-border)',
                background: 'var(--intent-color-surface)',
                fontSize: '0.875rem'
              }}
            >
              {item}
            </div>
          ))}
        </div>
      </div>
      <div style={{ padding: '8px 16px', fontSize: '0.75rem', color: 'var(--intent-color-muted)', borderTop: '1px solid var(--intent-color-border)' }}>
        Showing {visibleItems.length} of {items.length} items (virtualized)
      </div>
    </div>
  )
}

## Examples

### Large List

<div className="intent-demo">
  <VirtualList 
    items={Array.from({ length: 1000 }, (_, i) => `Item ${i + 1}`)}

```tsx
<VirtualList 
  items={largeArray}
  itemHeight={40}
  height={400}
  renderItem={(item) => <div>{item}</div>}
/>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `items` | `Array<any>` | `[]` | List items |
| `itemHeight` | `number` | `40` | Height of each item |
| `height` | `number` | `400` | Container height |
| `renderItem` | `(item, index) => ReactNode` | - | Item renderer |
| `overscan` | `number` | `5` | Extra items to render |
